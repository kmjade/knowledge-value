## 2. 核心模块原型

### 2.1 兼容层 — `CompatLayer.mqh`

这是双平台开发的基石。用预处理器宏统一 MQL4/MQL5 差异。

```cpp
// ──────────────────────────────────────────────────────────
// CompatLayer.mqh — MQL4/MQL5 兼容层
// ──────────────────────────────────────────────────────────
#ifndef __ENTERPY_COMPAT_LAYER__
#define __ENTERPY_COMPAT_LAYER__

// ----------------------------------------------------------
// 1. 平台检测
// ----------------------------------------------------------
#ifdef __MQL4__
   #define ENTERPY_PLATFORM_MQL4
#else
   #define ENTERPY_PLATFORM_MQL5
#endif

// ----------------------------------------------------------
// 2. 订单类型统一 (MQL4: OP_* / MQL5: POSITION_TYPE_*)
// ----------------------------------------------------------
#ifdef __MQL4__
   #define ENTERPY_OP_BUY       OP_BUY
   #define ENTERPY_OP_SELL      OP_SELL
   #define POSITION_TYPE_BUY    OP_BUY
   #define POSITION_TYPE_SELL   OP_SELL
#else
   #define ENTERPY_OP_BUY       POSITION_TYPE_BUY
   #define ENTERPY_OP_SELL      POSITION_TYPE_SELL
#endif

// ----------------------------------------------------------
// 3. 时间序列访问统一
//    MQL4: iClose(NULL, 0, i)
//    MQL5: CopyClose(_Symbol, _Period, i, 1, buf)
// ----------------------------------------------------------
#ifdef __MQL4__
   #define ENTERPY_OPEN(_i)    iOpen(_Symbol, PERIOD_CURRENT, _i)
   #define ENTERPY_HIGH(_i)    iHigh(_Symbol, PERIOD_CURRENT, _i)
   #define ENTERPY_LOW(_i)     iLow(_Symbol, PERIOD_CURRENT, _i)
   #define ENTERPY_CLOSE(_i)   iClose(_Symbol, PERIOD_CURRENT, _i)
   #define ENTERPY_VOLUME(_i)  iVolume(_Symbol, PERIOD_CURRENT, _i)
#else
   // MQL5 需要数组缓冲区
   extern double _enterpy_open[];
   extern double _enterpy_high[];
   extern double _enterpy_low[];
   extern double _enterpy_close[];
   extern long   _enterpy_volume[];
   #define ENTERPY_OPEN(_i)    _enterpy_open[_i]
   #define ENTERPY_HIGH(_i)    _enterpy_high[_i]
   #define ENTERPY_LOW(_i)     _enterpy_low[_i]
   #define ENTERPY_CLOSE(_i)   _enterpy_close[_i]
   #define ENTERPY_VOLUME(_i)  _enterpy_volume[_i]
#endif

// ----------------------------------------------------------
// 4. 指标调用统一
// ----------------------------------------------------------
#ifdef __MQL4__
   double ENTERPY_ATR(int period, int shift) {
      return iATR(_Symbol, PERIOD_CURRENT, period, shift);
   }
#else
   int    _atr_handle = INVALID_HANDLE;
   double _atr_buf[];

   double ENTERPY_ATR(int period, int shift) {
      if (_atr_handle == INVALID_HANDLE)
         _atr_handle = iATR(_Symbol, PERIOD_CURRENT, period);
      if (CopyBuffer(_atr_handle, 0, shift, 1, _atr_buf) > 0)
         return _atr_buf[0];
      return 0.0;
   }
#endif

// ----------------------------------------------------------
// 5. 持仓遍历统一
// ----------------------------------------------------------
#ifdef __MQL4__
   int ENTERPY_POSITIONS_TOTAL() {
      return OrdersTotal();
   }
   bool ENTERPY_SELECT_POSITION(int i) {
      return OrderSelect(i, SELECT_BY_POS, MODE_TRADES);
   }
   // ENTERPY 自己的持仓遍历（只扫当前品种+Magic）
   int ENTERPY_COUNT_MY_POSITIONS() {
      int count = 0;
      for (int i = OrdersTotal() - 1; i >= 0; i--) {
         if (OrderSelect(i, SELECT_BY_POS, MODE_TRADES)) {
            if (OrderSymbol() == _Symbol && OrderMagicNumber() == ENTERPY_MAGIC)
               count++;
         }
      }
      return count;
   }
#else
   int ENTERPY_POSITIONS_TOTAL() {
      return PositionsTotal();
   }
   bool ENTERPY_SELECT_POSITION(int i) {
      return PositionSelect(_Symbol);
   }
   int ENTERPY_COUNT_MY_POSITIONS() {
      int count = 0;
      for (int i = PositionsTotal() - 1; i >= 0; i--) {
         if (PositionSelect(_Symbol) && PositionGetInteger(POSITION_MAGIC) == ENTERPY_MAGIC)
            count++;
      }
      return count;
   }
#endif

#endif // __ENTERPY_COMPAT_LAYER__
```

### 2.3 策略接口 — `IStrategy.mqh`

```cpp
// ──────────────────────────────────────────────────────────
// IStrategy.mqh — 策略抽象接口（所有策略实现此接口）
// ──────────────────────────────────────────────────────────
#ifndef __ENTERPY_ISTRATEGY__
#define __ENTERPY_ISTRATEGY__

#include <Enterpy/Data/Indicators.mqh>
#include <Enterpy/Config.mqh>

class IStrategy {
protected:
   string   m_symbol;
   ENUM_TIMEFRAMES m_tf;
   double   m_entryPrice;       // 入场基准价（用于加仓计算）
   double   m_atrValue;         // 当前 ATR 值
   int      m_pyramidLevel;     // 当前加仓层级

public:
   // ─── 构造函数 ───
   IStrategy() {
      m_symbol   = _Symbol;
      m_tf       = PERIOD_CURRENT;
      m_pyramidLevel = 0;
      m_entryPrice   = 0.0;
      m_atrValue     = 0.0;
   }

   // ─── 纯虚接口（子类必须实现） ───
   virtual bool   ShouldEnterLong()  = 0;
   virtual bool   ShouldEnterShort() = 0;
   virtual bool   ShouldExitLong()   = 0;
   virtual bool   ShouldExitShort()  = 0;
   virtual double GetStopLoss(bool isLong) = 0;
   virtual string GetName()          = 0;

   // ─── 虚方法（可覆写） ───
   virtual bool ShouldAddToPosition(bool isLong) {
      if (m_pyramidLevel >= MaxUnits) return false;
      double step = m_atrValue * AddUnitStep;
      if (isLong && (Close(0) - m_entryPrice) >= step) return true;
      if (!isLong && (m_entryPrice - Close(0)) >= step) return true;
      return false;
   }

   // ─── 公开方法 ───
   void OnTick() {
      m_atrValue = CalcATR(ATR_Period, 1);
   }

   void SetEntryPrice(double price) {
      m_entryPrice = price;
   }

   void SetPyramidLevel(int level) {
      m_pyramidLevel = level;
   }

   int GetPyramidLevel() { return m_pyramidLevel; }

   void Reset() {
      m_pyramidLevel = 0;
      m_entryPrice   = 0.0;
   }
};

#endif // __ENTERPY_ISTRATEGY__
```

### 2.4 海龟 S1 — `TurtleS1.mqh`

```cpp
// ──────────────────────────────────────────────────────────
// TurtleS1.mqh — 海龟快速系统 (20日突破)
// ──────────────────────────────────────────────────────────
#ifndef __ENTERPY_TURTLE_S1__
#define __ENTERPY_TURTLE_S1__

#include <Enterpy/Strategy/IStrategy.mqh>
#include <Enterpy/Data/Indicators.mqh>

class TurtleS1 : public IStrategy {
private:
   int m_breakPeriod;       // 入场突破周期 (20)
   int m_exitPeriod;        // 出场突破周期 (10)

public:
   TurtleS1() {
      m_breakPeriod = Turtle1_Period;     // 20
      m_exitPeriod  = Turtle1_ExitPeriod; // 10
   }

   // ─── 入场: 价格突破 N日高点 ───
   bool ShouldEnterLong() override {
      double highest = DonchianHigh(m_breakPeriod, 1);
      // 突破前一日高点：Close[1] < Highest[1] 但当前突破
      double prevHigh = DonchianHigh(m_breakPeriod, 2);
      return (Close(1) < prevHigh) && (High(0) > highest);
   }

   // ─── 入场: 价格跌破 N日低点 ───
   bool ShouldEnterShort() override {
      double lowest  = DonchianLow(m_breakPeriod, 1);
      double prevLow = DonchianLow(m_breakPeriod, 2);
      return (Close(1) > prevLow) && (Low(0) < lowest);
   }

   // ─── 出场: 价格反向突破 M日低点 ───
   bool ShouldExitLong() override {
      double exitLevel = DonchianLow(m_exitPeriod, 1);
      return Low(0) <= exitLevel;
   }

   // ─── 出场: 价格反向突破 M日高点 ───
   bool ShouldExitShort() override {
      double exitLevel = DonchianHigh(m_exitPeriod, 1);
      return High(0) >= exitLevel;
   }

   // ─── 止损: 2 ATR ───
   double GetStopLoss(bool isLong) override {
      double atr = CalcATR(ATR_Period, 1);
      if (isLong) return Close(0) - (atr * StopLoss_ATR);
      else        return Close(0) + (atr * StopLoss_ATR);
   }

   string GetName() override { return "TurtleS1"; }
};

#endif // __ENTERPY_TURTLE_S1__
```

### 2.6 主 EA — `Enterpy.mq4`

```cpp
// ──────────────────────────────────────────────────────────
// Enterpy.mq4 — Enterpy Trade System 主入口 (MQL4)
// ──────────────────────────────────────────────────────────
#property copyright "Enterpy Trade System"
#property version   "0.10"
#property link      ""
#property strict

#include <Enterpy/Config.mqh>
#include <Enterpy/Data/Indicators.mqh>
#include <Enterpy/Strategy/TurtleS1.mqh>
#include <Enterpy/Strategy/TurtleS2.mqh>
#include <Enterpy/Execution/UnitSizing.mqh>
#include <Enterpy/Execution/OrderManager.mqh>
#include <Enterpy/Execution/PositionManager.mqh>
#include <Enterpy/Risk/RiskManager.mqh>
#include <Enterpy/Utils/Logger.mqh>

// ─── 全局对象 ───
TurtleS1        g_s1;
TurtleS2        g_s2;
RiskManager     g_risk;
OrderManager    g_orders;
PositionManager g_positions;
UnitSizing      g_sizing;
Logger          g_log("Enterpy");

int OnInit() {
   g_log.Info("Enterpy Trade System v0.1 初始化完成 (S1=%d, S2=%d)",
              Turtle1_Period, Turtle2_Period);
   return INIT_SUCCEEDED;
}

void OnDeinit(const int reason) {
   g_log.Info("Enterpy Trade System 关闭, reason=%d", reason);
}

// ──────────────────────────────────────────────────────────
// OnTick — 核心循环 (Tick 事件驱动)
// ──────────────────────────────────────────────────────────
void OnTick() {

   // ─── Step 1: 更新所有策略数据 ───
   g_s1.OnTick();
   g_s2.OnTick();

   // ─── Step 2: 风控检查 ───
   if (!g_risk.CheckSystem()) {
      // 风控触发：全平后停止
      if (g_positions.Count() > 0) {
         g_orders.CloseAll(ENTERPY_MAGIC, _Symbol);
         g_s1.Reset();
         g_s2.Reset();
      }
      return;
   }

   // ─── Step 3: 检查已持仓 → 出场 + 加仓 ───
   if (g_positions.Count() > 0) {
      ProcessExits();
      ProcessAdditions();
   }

   // ─── Step 4: 检查新入场 ───
   if (g_positions.Count() == 0) {
      ProcessEntries();
   }

   // ─── Step 5: 日志 ───
   g_log.Tick();
}

// ─── 出场逻辑 ───
void ProcessExits() {
   // S1 出场
   if (g_s1.ShouldExitLong())
      g_orders.CloseDirection(ENTERPY_MAGIC, _Symbol, ENTERPY_OP_BUY);
   if (g_s1.ShouldExitShort())
      g_orders.CloseDirection(ENTERPY_MAGIC, _Symbol, ENTERPY_OP_SELL);

   // S2 出场
   if (g_s2.ShouldExitLong())
      g_orders.CloseDirection(ENTERPY_MAGIC, _Symbol, ENTERPY_OP_BUY);
   if (g_s2.ShouldExitShort())
      g_orders.CloseDirection(ENTERPY_MAGIC, _Symbol, ENTERPY_OP_SELL);
}

// ─── 加仓逻辑 ───
void ProcessAdditions() {
   for (int i = 0; i < g_positions.Count(); i++) {
      if (!g_positions.SelectByIndex(i)) continue;

      bool isLong = (g_positions.Type() == ENTERPY_OP_BUY);
      int  level  = g_positions.GetPyramidLevel();
      double price = g_positions.OpenPrice();

      IStrategy *strategy = (g_positions.GetStrategyName() == "TurtleS1") ?
                              &g_s1 : &g_s2;

      strategy.SetEntryPrice(price);
      strategy.SetPyramidLevel(level);

      if (strategy.ShouldAddToPosition(isLong)) {
         double lots = g_sizing.CalculateUnits(price);
         if (lots > 0) {
            g_orders.OpenMarket(isLong ? ENTERPY_OP_BUY : ENTERPY_OP_SELL,
                                lots, strategy.GetName());
         }
      }
   }
}

// ─── 入场逻辑 ───
void ProcessEntries() {
   double lots;

   // S1 入场
   if (UseS1) {
      if (g_s1.ShouldEnterLong()) {
         lots = g_sizing.CalculateUnits(Close(0));
         g_orders.OpenMarket(ENTERPY_OP_BUY, lots, g_s1.GetName());
         g_s1.SetEntryPrice(Close(0));
      }
      else if (g_s1.ShouldEnterShort()) {
         lots = g_sizing.CalculateUnits(Close(0));
         g_orders.OpenMarket(ENTERPY_OP_SELL, lots, g_s1.GetName());
         g_s1.SetEntryPrice(Close(0));
      }
   }

   // S2 入场
   if (UseS2) {
      if (g_s2.ShouldEnterLong()) {
         lots = g_sizing.CalculateUnits(Close(0));
         g_orders.OpenMarket(ENTERPY_OP_BUY, lots, g_s2.GetName());
         g_s2.SetEntryPrice(Close(0));
      }
      else if (g_s2.ShouldEnterShort()) {
         lots = g_sizing.CalculateUnits(Close(0));
         g_orders.OpenMarket(ENTERPY_OP_SELL, lots, g_s2.GetName());
         g_s2.SetEntryPrice(Close(0));
      }
   }
}
```

### 2.7 仓位计算 — `UnitSizing.mqh`

```cpp
// ──────────────────────────────────────────────────────────
// UnitSizing.mqh — 海龟单位计算
// 1 Unit = 账户余额 × 风险比率 ÷ (ATR × 每点价值)
// ──────────────────────────────────────────────────────────
#ifndef __ENTERPY_UNIT_SIZING__
#define __ENTERPY_UNIT_SIZING__

#include <Enterpy/Data/Indicators.mqh>

class UnitSizing {
public:
   double CalculateUnits(double entryPrice) {
      double accountRisk   = AccountBalance() * Risk_Percent / 100.0;
      double atrValue      = CalcATR(ATR_Period, 1);
      double pointValue    = MarketInfo(_Symbol, MODE_TICKVALUE);
      double riskPerUnit   = atrValue * pointValue;

      if (riskPerUnit <= 0) return 0.0;

      double rawUnits = accountRisk / riskPerUnit;

      // 手数规范化
      double lotStep  = MarketInfo(_Symbol, MODE_LOTSTEP);
      double minLot   = MarketInfo(_Symbol, MODE_MINLOT);
      double maxLot   = MarketInfo(_Symbol, MODE_MAXLOT);

      double lots = MathFloor(rawUnits / lotStep) * lotStep;
      lots = MathMax(minLot, MathMin(maxLot, lots));

      return lots;
   }
};

#endif // __ENTERPY_UNIT_SIZING__
```

## 4. 风控引擎原型

### 4.1 `RiskManager.mqh`

```cpp
#ifndef __ENTERPY_RISK_MANAGER__
#define __ENTERPY_RISK_MANAGER__

#include <Enterpy/Config.mqh>
#include <Enterpy/Data/Indicators.mqh>

class RiskManager {
private:
   double m_startBalance;      // 初始余额
   double m_peakBalance;       // 历史峰值
   bool   m_haltTrading;       // 交易暂停标志

public:
   RiskManager() {
      m_startBalance = AccountBalance();
      m_peakBalance  = AccountBalance();
      m_haltTrading  = false;
   }

   // 检查系统是否可交易
   bool CheckSystem() {
      double currentBalance = AccountBalance();
      double equity         = AccountEquity();

      // 更新峰值
      if (equity > m_peakBalance)
         m_peakBalance = equity;

      // 计算当前回撤
      double drawdown = 0;
      if (m_peakBalance > 0)
         drawdown = (m_peakBalance - equity) / m_peakBalance * 100.0;

      // ─── 回撤超过最大允许 → 暂停交易 ───
      if (drawdown >= Max_DD_Percent) {
         if (!m_haltTrading) {
            Print("[RISK] 回撤 ", DoubleToString(drawdown, 1),
                  "% 超过阈值 ", DoubleToString(Max_DD_Percent, 1),
                  "%，交易暂停!");
            m_haltTrading = true;
         }
         return false;  // 平仓退出
      }

      // ─── 回撤恢复到安全区 → 恢复交易 ───
      if (m_haltTrading && drawdown <= Resume_DD_Percent) {
         Print("[RISK] 回撤恢复至 ", DoubleToString(drawdown, 1),
               "%，交易恢复");
         m_haltTrading = false;
      }

      return !m_haltTrading;
   }

   bool IsHalted() { return m_haltTrading; }
};

#endif // __ENTERPY_RISK_MANAGER__
```

### 4.2 风控矩阵

| 检查项 | 触发条件 | 动作 | 自动恢复 |
|:-------|:---------|:-----|:---------|
| 净值回撤 | DD ≥ Max_DD_Percent (20%) | 全平 → 暂停交易 | DD ≤ Resume_DD_Percent (10%) |
| 硬止损 | 价格反向 2 ATR | 该单位平仓 | — |
| 盈亏平衡止损 | 盈利 ≥ 1 ATR | SL 移至入场价 | — |
| 追踪止损 | 盈利每 +1 ATR | SL 上移 ½ ATR | — |
| 最大持仓 | 总单位 ≥ MaxUnits×2 (8) | 拒绝新开仓 | 持仓减少后 |

## 6. 风险与注意事项

### 6.1 MQL4 已知局限

| 问题 | 影响 | 应对 |
|:-----|:-----|:-----|
| 历史订单数量限制 | 长周期回测可能丢失早期数据 | 定期归档 |
| `OrderSelect()` 循环中修改订单 | 索引移位导致遗漏 | 逆序遍历 |
| 点差变化 | 回测与实盘执行差异 | 回测时用实际点差 |
| 经纪商 GMT 偏移 | 日线收盘时间不同 | 兼容层中处理 |

### 6.2 海龟策略固有风险

| 风险 | 说明 | 缓解 |
|:-----|:------|:-----|
| 大幅回撤期 | 趋势跟踪在震荡市中连续亏损 | 风控引擎暂停机制 |
| 滑点冲击 | 突破入场在极端行情中滑点大 | 使用限价单替代市价单（可选） |
| 参数过拟合 | 优化后参数在样本外失效 | 固定经典参数 |
| 流动性枯竭 | 小品种突破时流动性不足 | 过滤低流动性品种 |
