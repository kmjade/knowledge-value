# ============ 特殊限制规则 / Special Restriction Rules ============

# IoT 只能访问互联网
config rule
    option name 'Allow-IoT-to-Internet'
    option src 'iot'
    option dest 'wan'
    option proto 'all'
    option target 'ACCEPT'

# IoT 设备间隔离
config rule
    option name 'Block-IoT-to-IoT'
    option src 'iot'
    option dest 'iot'
    option proto 'all'
    option target 'REJECT'

# Main 到 IoT 只读 (示例: 阻止写操作)
config rule
    option name 'Main-to-IoT-Read-Only'

    option dest 'iot'

    option dest_port '80 443'
    option target 'ACCEPT'

# Guest 带宽限制
config rule
    option name 'Guest-Bandwidth-Limit'
    option src 'guest'
    option dest '*'
    option proto 'all'
    option target 'ACCEPT'
    option extra '-m limit --limit 50/second'
