                const int prev_calculated,
                const datetime &time[],
                const double &open[],
                const double &high[],
                const double &low[],
                const double &close[],
                const long &tick_volume[],
                const long &volume[],
                const int &spread[])

      return(0);
   
   int start = prev_calculated;
   if(start == 0)
      start = RSIPeriod;
   
   // è¨ˆç®—RSI
   for(int i = start; i < rates_total; i++)
     {
      double gains = 0;
      double losses = 0;
      
      // è¨ˆç®—åˆå§‹å¹³å‡æ”¶ç›Šå’ŒæŸå¤±
      for(int j = i - RSIPeriod + 1; j <= i; j++)

         double rs = avgGain / avgLoss;

        }

      OversoldBuffer[i] = Oversold;
