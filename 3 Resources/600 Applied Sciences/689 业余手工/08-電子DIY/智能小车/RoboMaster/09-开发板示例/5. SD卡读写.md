# SD卡读写

> [!summary] 概述
> STM32 开发板 SD 卡读写示例，集成 FatFS 文件系统，支持文件的创建、读取、写入等操作。

---

## 功能说明

### 核心功能

- SD 卡读写操作
- FatFS 文件系统支持
- 文件创建、读取、写入、删除
- 目录管理

### 应用场景

| 场景 | 说明 |
|------|------|
| **数据记录** | 记录传感器数据、运行日志 |
| **参数存储** | 保存配置文件 |
| **固件升级** | 存放升级文件 |
| **音频播放** | 存储音频文件 |

---

## 技术架构

### SDIO 接口

| 参数 | 值 |
|------|-----|
| **接口** | SDIO (Secure Digital Input Output) |
| **时钟** | 最高 48MHz |
| **数据位宽** | 1-bit / 4-bit |
| **传输模式** | SD/SDHC/SDXC |

### FatFS 文件系统

FatFS 是专为小型嵌入式系统设计的 FAT 文件系统：

| 特性 | 说明 |
|------|------|
| **支持格式** | FAT12 / FAT16 / FAT32 / exFAT |
| **代码体积** | 小巧，适合资源受限系统 |
| **可移植性** | 纯 C 语言，平台无关 |
| **开源** | 免费，MIT 协议 |

---

## 硬件接口

### SDIO 引脚

| 信号 | GPIO | 说明 |
|------|------|------|
| SDIO_CK | PC12 | 时钟 |
| SDIO_CMD | PD2 | 命令 |
| SDIO_D0 | PC8 | 数据线 0 |
| SDIO_D1 | PC9 | 数据线 1 |
| SDIO_D2 | PC10 | 数据线 2 |
| SDIO_D3 | PC11 | 数据线 3 |

### 卡槽规格

- 支持 SD 卡、SDHC 卡、SDXC 卡
- 支持 SPI 模式和 SDIO 模式
- 支持热插拔检测（CD 引脚）

## 使用示例

### 挂载 SD 卡

```c
#include "ff.h"
#include "fatfs.h"

FATFS fs;    // 文件系统对象
FRESULT res;

// 挂载 SD 卡
void SD_Mount(void)
{
    res = f_mount(&fs, "0:", 1);
    if (res != FR_OK) {
        printf("Mount failed: %d\n", res);
        return;
    }
    printf("SD card mounted!\n");
}

// 卸载 SD 卡
void SD_Unmount(void)
{
    f_mount(NULL, "0:", 0);
    printf("SD card unmounted!\n");
}
```

### 文件写入

```c
FIL file;
UINT bytes_written;

void SD_WriteLog(const char *data)
{
    // 打开文件（追加模式）
    res = f_open(&file, "0:/log.txt", FA_WRITE | FA_OPEN_APPEND);
    if (res != FR_OK) {
        // 文件不存在，创建新文件
        res = f_open(&file, "0:/log.txt", FA_WRITE | FA_CREATE_NEW);
        if (res != FR_OK) {
            printf("Create file failed: %d\n", res);
            return;
        }
    }
    
    // 写入数据
    res = f_write(&file, data, strlen(data), &bytes_written);
    if (res != FR_OK) {
        printf("Write failed: %d\n", res);
    }
    
    // 同步并关闭
    f_sync(&file);
    f_close(&file);
}
```

### 文件读取

```c
void SD_ReadFile(const char *filename)
{
    FIL file;
    UINT bytes_read;
    char buffer[128];
    
    // 打开文件
    res = f_open(&file, filename, FA_READ);
    if (res != FR_OK) {
        printf("Open failed: %d\n", res);
        return;
    }
    
    // 读取文件内容
    while (1) {
        res = f_read(&file, buffer, sizeof(buffer) - 1, &bytes_read);
        if (res != FR_OK || bytes_read == 0) {
            break;
        }
        buffer[bytes_read] = '\0';
        printf("%s", buffer);
    }
    
    // 关闭文件
    f_close(&file);
}
```

### 目录遍历

```c
void SD_ListFiles(const char *path)
{
    DIR dir;
    FILINFO fno;
    
    res = f_opendir(&dir, path);
    if (res != FR_OK) {
        printf("Open directory failed: %d\n", res);
        return;
    }
    
    printf("Files in %s:\n", path);
    
    while (1) {
        res = f_readdir(&dir, &fno);
        if (res != FR_OK || fno.fname[0] == 0) {
            break;
        }
        
        if (fno.fattrib & AM_DIR) {
            printf("  [DIR]  %s\n", fno.fname);
        } else {
            printf("  [FILE] %s (%lu bytes)\n", fno.fname, fno.fsize);
        }
    }
    
    f_closedir(&dir);
}
```

### 配置文件读写

```c
// 保存参数到配置文件
void SaveConfig(int param1, float param2)
{
    FIL file;
    char buffer[64];
    UINT bytes_written;
    
    sprintf(buffer, "param1=%d\nparam2=%.2f\n", param1, param2);
    
    f_open(&file, "0:/config.ini", FA_WRITE | FA_CREATE_ALWAYS);
    f_write(&file, buffer, strlen(buffer), &bytes_written);
    f_close(&file);
}

// 读取配置文件
int LoadConfig(int *param1, float *param2)
{
    FIL file;
    char buffer[64];
    UINT bytes_read;
    
    if (f_open(&file, "0:/config.ini", FA_READ) != FR_OK) {
        return -1;  // 文件不存在
    }
    
    f_read(&file, buffer, sizeof(buffer), &bytes_read);
    buffer[bytes_read] = '\0';
    
    sscanf(buffer, "param1=%d\nparam2=%f\n", param1, param2);
    
    f_close(&file);
    return 0;
}
```

## 数据记录示例

### 循环缓冲记录

```c
#define LOG_FILE "0:/data/log.csv"
#define MAX_LOG_SIZE (1024 * 1024)  // 1MB

void LogSensorData(int sensor1, int sensor2, int sensor3)
{
    static FIL file;
    static int initialized = 0;
    char buffer[64];
    UINT bytes_written;
    
    if (!initialized) {
        f_open(&file, LOG_FILE, FA_WRITE | FA_OPEN_APPEND);
        initialized = 1;
    }
    
    // 格式化数据
    sprintf(buffer, "%lu,%d,%d,%d\n", 
            HAL_GetTick(), sensor1, sensor2, sensor3);
    
    f_write(&file, buffer, strlen(buffer), &bytes_written);
    
    // 定期同步
    static int sync_counter = 0;
    if (++sync_counter >= 100) {
        f_sync(&file);
        sync_counter = 0;
    }
}
```

## FatFS 配置

### ffconf.h 配置

| 配置项 | 推荐值 | 说明 |
|--------|--------|------|
| FF_FS_READONLY | 0 | 启用写操作 |
| FF_FS_MINIMIZE | 0 | 完整功能 |
| FF_USE_STRFUNC | 1 | 启用字符串函数 |
| FF_USE_FIND | 1 | 启用查找功能 |
| FF_USE_MKFS | 1 | 启用格式化 |
| FF_USE_FASTSEEK | 1 | 启用快速定位 |
| FF_CODE_PAGE | 936 | 简体中文 |
| FF_USE_LFN | 1 | 长文件名支持 |
| FF_MAX_LFN | 255 | 最大文件名长度 |

## 调试方法

### SD 卡检测

```c
// 检测 SD 卡是否插入
uint8_t SD_IsDetected(void)
{
    // 检查 CD (Card Detect) 引脚
    return HAL_GPIO_ReadPin(SD_CD_GPIO_Port, SD_CD_Pin) == GPIO_PIN_RESET;
}
```

### 容量查询

```c
void SD_GetCapacity(void)
{
    FATFS *fs;
    DWORD fre_clust, fre_sect, tot_sect;
    
    f_getfree("0:", &fre_clust, &fs);
    
    tot_sect = (fs->n_fatent - 2) * fs->csize;
    fre_sect = fre_clust * fs->csize;
    
    printf("Total: %lu MB\n", tot_sect / 2048);
    printf("Free: %lu MB\n", fre_sect / 2048);
}
```

### 常见问题

| 问题 | 可能原因 | 解决方法 |

| 写入失败 | 写保护 | 检查写保护开关 |
| 读取错误 | 文件系统损坏 | 格式化 SD 卡 |
