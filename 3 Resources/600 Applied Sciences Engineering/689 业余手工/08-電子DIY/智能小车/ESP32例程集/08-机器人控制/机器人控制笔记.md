## 🤖 差速运动学

### 运动学模型

```
        ↑ v_linear (线速度)
        │
    ┌───┴───┐
    │       │
    │   ●   │ → ω_angular (角速度)
    │       │
    └───┬───┘
        │
        ▼
```

### 前向运动学

给定左右轮速度，计算机器人线速度和角速度：

```python
def forward_kinematics(v_left, v_right, wheelbase):
    """
    前向运动学
    :param v_left: 左轮速度 (m/s)
    :param v_right: 右轮速度 (m/s)
    :param wheelbase: 轮距 (m)
    :return: 线速度, 角速度
    """
    v_linear = (v_left + v_right) / 2
    ω_angular = (v_right - v_left) / wheelbase

    return v_linear, ω_angular
```

### 逆向运动学

给定线速度和角速度，计算左右轮速度：

```python
def inverse_kinematics(v_linear, ω_angular, wheelbase):
    """
    逆向运动学
    :param v_linear: 线速度 (m/s)
    :param ω_angular: 角速度 (rad/s)
    :param wheelbase: 轮距 (m)
    :return: 左轮速度, 右轮速度
    """
    v_left = v_linear - (ω_angular * wheelbase) / 2
    v_right = v_linear + (ω_angular * wheelbase) / 2

    return v_left, v_right
```

## ⚙️ PID控制

### PID控制器原理

```python
class PIDController:
    def __init__(self, kp, ki, kd, output_limits=(None, None)):
        """
        PID控制器
        :param kp: 比例增益
        :param ki: 积分增益
        :param kd: 微分增益
        :param output_limits: 输出限制 (min, max)
        """
        self.kp = kp
        self.ki = ki
        self.kd = kd
        self.output_limits = output_limits

        self.prev_error = 0
        self.integral = 0

    def compute(self, setpoint, measured_value, dt):
        """
        计算PID输出
        :param setpoint: 目标值
        :param measured_value: 测量值
        :param dt: 时间步长 (s)
        :return: 控制输出
        """
        error = setpoint - measured_value

        # 比例项
        p_term = self.kp * error

        # 积分项
        self.integral += error * dt
        i_term = self.ki * self.integral

        # 微分项
        derivative = (error - self.prev_error) / dt
        d_term = self.kd * derivative

        # PID输出
        output = p_term + i_term + d_term

        # 限制输出范围
        if self.output_limits[0] is not None:
            output = max(self.output_limits[0], output)
        if self.output_limits[1] is not None:
            output = min(self.output_limits[1], output)

        self.prev_error = error
        return output

    def reset(self):
        """重置控制器状态"""
        self.prev_error = 0
        self.integral = 0
```

### 速度控制示例

```python
# 速度PID控制器
velocity_pid = PIDController(
    kp=1.0,   # 比例增益
    ki=0.5,   # 积分增益
    kd=0.1,   # 微分增益
    output_limits=(-1.0, 1.0)  # PWM输出范围 -1.0~1.0
)

# 控制循环
def velocity_control_loop(target_velocity, current_velocity, dt):
    # 计算PID输出
    control_output = velocity_pid.compute(target_velocity, current_velocity, dt)

    # 将控制输出转换为PWM信号
    pwm = control_output * 255  # 假设PWM范围 0-255

    return pwm
```

## 🧭 状态估计

### 编程器里程计

```python
import numpy as np

class Odometry:
    def __init__(self, wheelbase, wheel_radius, ticks_per_rev):
        """
        里程计状态估计
        :param wheelbase: 轮距 (m)
        :param wheel_radius: 轮子半径 (m)
        :param ticks_per_rev: 编码器每圈脉冲数
        """
        self.wheelbase = wheelbase
        self.wheel_radius = wheel_radius
        self.ticks_per_rev = ticks_per_rev

        # 机器人位姿 [x, y, theta]
        self.pose = np.array([0.0, 0.0, 0.0])

        # 上一时刻编码器值
        self.prev_left_ticks = 0
        self.prev_right_ticks = 0

    def ticks_to_meters(self, ticks):
        """将编码器脉冲数转换为米"""
        meters_per_tick = (2 * np.pi * self.wheel_radius) / self.ticks_per_rev
        return ticks * meters_per_tick

    def update(self, left_ticks, right_ticks, dt):
        """
        更新里程计
        :param left_ticks: 左轮编码器值
        :param right_ticks: 右轮编码器值
        :param dt: 时间步长 (s)
        """
        # 计算编码器增量
        delta_left = left_ticks - self.prev_left_ticks
        delta_right = right_ticks - self.prev_right_ticks

        # 转换为距离
        dist_left = self.ticks_to_meters(delta_left)
        dist_right = self.ticks_to_meters(delta_right)

        # 更新上一时刻编码器值
        self.prev_left_ticks = left_ticks
        self.prev_right_ticks = right_ticks

        # 计算线速度和角速度
        v_linear = (dist_left + dist_right) / (2 * dt)
        ω_angular = (dist_right - dist_left) / (self.wheelbase * dt)

        # 更新位姿（假设匀速运动）
        x, y, theta = self.pose

        # 位置更新
        if abs(ω_angular) < 0.001:  # 直线运动
            x += v_linear * np.cos(theta) * dt
            y += v_linear * np.sin(theta) * dt
        else:  # 圆弧运动
            x += (v_linear / ω_angular) * (np.sin(theta + ω_angular * dt) - np.sin(theta))
            y += (v_linear / ω_angular) * (-np.cos(theta + ω_angular * dt) + np.cos(theta))
            theta += ω_angular * dt

        self.pose = np.array([x, y, theta])
        return v_linear, ω_angular, self.pose
```

## 🧱 IMU数据融合

### 姿态估计

```python
class IMUFusion:
    def __init__(self):
        """IMU数据融合"""
        self.roll = 0.0
        self.pitch = 0.0
        self.yaw = 0.0

    def update_accelerometer(self, ax, ay, az):
        """
        从加速度计计算roll和pitch
        :param ax, ay, az: 加速度计数据 (m/s^2)
        """
        self.roll = np.arctan2(ay, az)
        self.pitch = np.arctan2(-ax, np.sqrt(ay**2 + az**2))

    def update_gyroscope(self, gx, gy, gz, dt):
        """
        从陀螺仪积分计算角速度
        :param gx, gy, gz: 陀螺仪数据 (rad/s)
        :param dt: 时间步长 (s)
        """
        self.roll += gx * dt
        self.pitch += gy * dt
        self.yaw += gz * dt

    def complementary_filter(self, ax, ay, az, gx, gy, gz, dt, alpha=0.95):
        """
        互补滤波器融合加速度计和陀螺仪
        :param alpha: 陀螺仪权重 (0-1)
        """
        # 加速度计计算roll和pitch
        accel_roll = np.arctan2(ay, az)
        accel_pitch = np.arctan2(-ax, np.sqrt(ay**2 + az**2))

        # 互补滤波
        self.roll = alpha * (self.roll + gx * dt) + (1 - alpha) * accel_roll
        self.pitch = alpha * (self.pitch + gy * dt) + (1 - alpha) * accel_pitch
        self.yaw += gz * dt  # 偏航角仅使用陀螺仪
```

## 🎯 标定

### 速度标定

```python
class VelocityCalibration:
    def __init__(self):
        self.measurements = []

    def collect(self, pwm_command, actual_velocity):
        """
        收集标定数据
        :param pwm_command: PWM命令值
        :param actual_velocity: 实际速度 (m/s)
        """
        self.measurements.append({
            'pwm': pwm_command,
            'velocity': actual_velocity
        })

    def fit_model(self):
        """
        拟合PWM-速度模型
        :return: 线性模型参数 (slope, intercept)
        """
        import numpy as np

        # 提取数据
        pwm = np.array([m['pwm'] for m in self.measurements])
        velocity = np.array([m['velocity'] for m in self.measurements])

        # 线性拟合: velocity = slope * pwm + intercept
        coeffs = np.polyfit(pwm, velocity, 1)
        slope, intercept = coeffs

        return slope, intercept

    def pwm_to_velocity(self, pwm, slope, intercept):
        """PWM转速度"""
        return slope * pwm + intercept

    def velocity_to_pwm(self, velocity, slope, intercept):
        """速度转PWM"""
        return (velocity - intercept) / slope
```

## 📡 传感器集成

### 激光雷达

```python
from sensor_msgs.msg import LaserScan

def scan_callback(msg):
    """
    激光雷达数据处理
    """
    angle_min = msg.angle_min
    angle_increment = msg.angle_increment
    ranges = msg.ranges

    # 处理点云数据
    points = []
    for i, range_val in enumerate(ranges):
        if range_val < msg.range_max and range_val > msg.range_min:
            angle = angle_min + i * angle_increment
            x = range_val * np.cos(angle)
            y = range_val * np.sin(angle)
            points.append([x, y])

    return points
```

### TF2使用

```python
import rclpy
from rclpy.node import Node
from tf2_ros import TransformListener, Buffer
from tf2_ros import TransformException

class TFListener(Node):
    def __init__(self):
        super().__init__('tf_listener')
        self.tf_buffer = Buffer()
        self.tf_listener = TransformListener(self.tf_buffer, self)
        self.timer = self.create_timer(0.1, self.on_timer)

    def on_timer(self):
        try:
            # 获取坐标变换
            transform = self.tf_buffer.lookup_transform(
                'base_link',
                'laser_link',
                rclpy.time.Time())

            self.get_logger().info(
                f'Transform: {transform.transform.translation.x}, '
                f'{transform.transform.translation.y}, '
                f'{transform.transform.translation.z}')

        except TransformException as ex:
            self.get_logger().error(f'TF error: {ex}')
```
