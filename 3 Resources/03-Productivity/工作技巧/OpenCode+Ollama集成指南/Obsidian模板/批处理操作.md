### 選擇范围 / Selection Scope

<%*
const scopes = [
    '📁 当前資料夾 - Current folder',
    '🌐 整个仓库 - Entire vault', 
    '🏷️ 按標籤篩選 - Filter by tag',
    '📅 按日期篩選 - Filter by date',
    '🔤 按檔案类型篩選 - Filter by file type',
    '📋 自定义檔案列表 - Custom file list'
];

const scopeValues = ['folder', 'vault', 'tag', 'date', 'filetype', 'custom'];

const selectedScope = await tp.system.suggester(
    scopes,
    scopeValues,
    false,
    '選擇處理范围 / Select processing scope:'
);

let filterCriteria = "";
let dataviewQuery = "";

// 根据選擇范围設置查詢条件
switch (selectedScope) {
    case 'folder':
        const currentFolder = tp.file.folder;
        dataviewQuery = `FROM "${currentFolder}"`;
        break;
        
    case 'tag':
        filterCriteria = await tp.system.prompt("輸入標籤 / Enter tag:");
        dataviewQuery = `FROM #${filterCriteria}`;
        break;
        
    case 'date':
        filterCriteria = await tp.system.prompt("輸入日期范围 (YYYY-MM-DD) / Enter date range:");
        const dateRange = filterCriteria.includes('..') ? filterCriteria : `>= ${filterCriteria}`;
        dataviewQuery = `FROM "" WHERE file.mday >= date("${dateRange}")`;
        break;
        
    case 'filetype':
        const fileTypes = ['.md', '.py', '.js', '.ts', '.java', '.cpp', '.go'];
        const selectedType = await tp.system.suggester(fileTypes, fileTypes, false, '選擇檔案类型:');
        filterCriteria = selectedType;
        dataviewQuery = `WHERE contains(file.path, "${selectedType}")`;
        break;
        
    case 'custom':
        const customFiles = await tp.system.prompt("輸入檔案路徑列表 (用逗号分隔) / Enter file paths (comma-separated):");
        filterCriteria = customFiles;
        break;
        
    default:
        dataviewQuery = '';
}
-%>

## 📋 生成的批處理腳本 / Generated Batch Script

<%*
// 根据操作类型生成腳本
let scriptContent = '';
let scriptName = '';
let fileExtension = '.sh'; // 默认为shell腳本

switch (selectedBatchType) {
    case 'file-analysis':
# 分析
        scriptContent = `#!/bin/bash
# 分析
# Generated by OpenCode + Ollama Integration

set -e  # 遇到错误时退出

# 分析
# 配置
echo "   模型: ${selectedModel}"
echo "   批大小: ${batchSizeNum}"
echo "   并发数: ${concurrencyNum}"
echo ""

# 處理函数
process_file() {
    local file="$1"
    local batch_id="$2"
    
# 分析
    
# 分析
# 分析
        --model ollama/${selectedModel} \\
        --timeout 180 \\
        "$file" > "${file}.analysis.md" 2>&1
    
    if [ $? -eq 0 ]; then
        echo "✅ [批${batch_id}] 完成: $file"
    else
        echo "❌ [批${batch_id}] 失败: $file"
        echo "   错误資訊已儲存到 ${file}.error.log"
    fi
    
    echo ""
}

# 主處理逻辑
batch_id=1
file_count=0

${selectedScope === 'custom' ? `
# 處理自定义檔案列表
files="${filterCriteria}"
for file in \$(echo "$files" | tr ',' '\\n'); do
    if [ -f "$file" ]; then
        process_file "$file" "$batch_id"
        
        file_count=$((file_count + 1))
        if [ $((file_count % ${batchSizeNum})) -eq 0 ]; then
            batch_id=$((batch_id + 1))
            
            # 等待当前批次完成
            wait
            echo "⏳ 等待下一批次..."
            sleep 5
        fi
        
        # 并发控制
        if [ $((file_count % ${concurrencyNum})) -eq 0 ]; then
            wait
        fi
    fi
done
` : `
# 使用find尋找檔案
find_command="find . -name '*.md' -type f"

${selectedScope === 'folder' ? `find_command="find '${tp.file.folder}' -name '*.md' -type f"` : ''}
${selectedScope === 'tag' ? `# 根据標籤篩選需要手動實現
echo "⚠️ 基于標籤的篩選请手動提供檔案列表"
exit 1` : ''}
${selectedScope === 'date' ? `find_command="find . -name '*.md' -type f -newermt '${filterCriteria}'"` : ''}
${selectedScope === 'filetype' ? `find_command="find . -name '${filterCriteria}' -type f"` : ''}

# 执行尋找和處理
eval "$find_command" | head -$((${batchSizeNum} * 10)) | while read file; do
    if [ -f "$file" ]; then
        process_file "$file" "$batch_id"
        
        file_count=$((file_count + 1))
        if [ $((file_count % ${batchSizeNum})) -eq 0 ]; then
            batch_id=$((batch_id + 1))
            
            # 等待当前批次完成
            wait
            echo "⏳ 等待下一批次..."
            sleep 5
        fi
        
        # 并发控制
        if [ $((file_count % ${concurrencyNum})) -eq 0 ]; then
            wait
        fi
    fi
done
`}

# 等待所有任務完成
wait

echo ""
# 分析
echo "📊 處理統計:"
echo "   處理檔案数: $file_count"
echo "   生成報告数: $file_count"
echo "   報告位置: 各檔案同目錄的 .analysis.md"
`;
        break;
        
    case 'tag-generation':
        scriptName = '批量標籤生成';
        scriptContent = `#!/bin/bash
# 批量標籤生成腳本 / Bulk Tag Generation Script
# Generated by OpenCode + Ollama Integration

set -e

echo "🏷️ 開始批量標籤生成..."
# 配置
echo "   模型: ${selectedModel}"
echo "   批大小: ${batchSizeNum}"
echo "   并发数: ${concurrencyNum}"
echo ""

# 處理函数
process_file() {
    local file="$1"
    local batch_id="$2"
    
    echo "📁 [批${batch_id}] 生成標籤: $file"
    
    # 讀取檔案內容
    content=\$(cat "$file")
    
    # 使用OpenCode生成標籤
    tags=\$(opencode run "基于這個檔案的內容，生成5-10个相關的標籤，用逗号分隔，只返回標籤列表" \\
        --model ollama/${selectedModel} \\
        --timeout 120 \\
        "$file" 2>/dev/null | grep -o "[^,]*" | tr '\\n' ',' | sed 's/,$//')
    
    if [ $? -eq 0 ] && [ -n "$tags" ]; then
# 更新
        temp_file="\${file}.tmp"
        
        # 檢查是否已有frontmatter
        if grep -q "^---" "$file"; then
            # 已有frontmatter，新增標籤
            sed -i "/^tags:/a\\  - $tags" "$file"
        else
            # 新增frontmatter和標籤
            cat > "$temp_file" << EOF
---
tags:
  - $tags
---

$content
EOF
            mv "$temp_file" "$file"
        fi
        
        echo "✅ [批${batch_id}] 標籤已新增: $tags"
    else
        echo "❌ [批${batch_id}] 標籤生成失败: $file"
    fi
    
    echo ""
}

# 分析
${selectedScope === 'custom' ? `
files="${filterCriteria}"
for file in \$(echo "$files" | tr ',' '\\n'); do
    if [ -f "$file" ]; then
        process_file "$file" "1"
    fi
done
` : `
find . -name '*.md' -type f | head -$((${batchSizeNum} * 10)) | while read file; do
    process_file "$file" "1"
done
`}

wait
echo "🎉 批量標籤生成完成！"
`;
        break;
        
    case 'code-refactoring':
        scriptName = '批量代碼重构';
        scriptContent = `#!/bin/bash
# 批量代碼重构腳本 / Bulk Code Refactoring Script
# Generated by OpenCode + Ollama Integration

set -e

echo "🛠️ 開始批量代碼重构..."
# 配置
echo "   模型: ${selectedModel}"
echo "   批大小: ${batchSizeNum}"
echo ""

# 處理函数
process_file() {
    local file="$1"
    local batch_id="$2"
    
    echo "🔧 [批${batch_id}] 重构檔案: $file"
    
    # 創建備份
    cp "$file" "${file}.backup"
    
    # 使用OpenCode重构代碼
    opencode run "重构這個檔案中的代碼，改进可读性、效能和可維護性，保持功能不变，直接輸出重构后的完整代碼" \\
        --model ollama/${selectedModel} \\
        --timeout 300 \\
        "$file" > "${file}.refactored" 2>&1
    
    if [ $? -eq 0 ]; then
        # 驗證重构結果
        if [ -s "${file}.refactored" ]; then
            mv "${file}.refactored" "$file"
            echo "✅ [批${batch_id}] 重构完成: $file"
        else
            echo "❌ [批${batch_id}] 重构結果为空: $file"
            mv "${file}.backup" "$file"
        fi
    else
        echo "❌ [批${batch_id}] 重构失败: $file"
        mv "${file}.backup" "$file"
        rm -f "${file}.refactored"
    fi
    
    echo ""
}

# 處理代碼檔案
find . -name '*.py' -o -name '*.js' -o -name '*.ts' -o -name '*.java' | head -$((${batchSizeNum} * 5)) | while read file; do
    process_file "$file" "1"
done

wait
echo "🎉 批量代碼重构完成！"
echo "📊 備份檔案位置: *.backup"
`;
        break;
        
    default:
        scriptName = '通用批處理腳本';
        scriptContent = `#!/bin/bash
# 通用批處理腳本 / Generic Batch Processing Script
# Generated by OpenCode + Ollama Integration

set -e

echo "⚡ 開始批處理操作..."
# 配置
echo "   操作类型: ${selectedBatchType}"
echo "   模型: ${selectedModel}"
echo "   批大小: ${batchSizeNum}"
echo ""

# 通用處理函数
process_file() {
    local file="$1"
    local batch_id="$2"
    
    echo "📁 [批${batch_id}] 處理檔案: $file"
    
    # 根据操作类型执行不同的OpenCode命令
    case "${selectedBatchType}" in
        "data-extraction")
            opencode run "从檔案中提取關鍵數據和資訊，结构化輸出" --model ollama/${selectedModel} "$file"
            ;;
        "documentation-generation")
            opencode run "为這個檔案生成完整的文檔說明" --model ollama/${selectedModel} "$file"
            ;;
        "test-generation")
            opencode run "为這個代碼檔案生成全面的单元測試" --model ollama/${selectedModel} "$file"
            ;;
        "format-conversion")
            opencode run "将檔案內容转换为指定格式" --model ollama/${selectedModel} "$file"
            ;;
        *)
            opencode run "对檔案执行${selectedBatchType}操作" --model ollama/${selectedModel} "$file"
            ;;
    esac
    
    echo "✅ [批${batch_id}] 處理完成: $file"
    echo ""
}

# 执行處理
find . -name '*.md' -type f | head -$((${batchSizeNum} * 5)) | while read file; do
    process_file "$file" "1"
done

wait
echo "🎉 批處理操作完成！"
`;
}

// 設置腳本檔案名
scriptFileName = `${tp.date.now('YYYY-MM-DD')}-${scriptName.replaceAll(' ', '-')}.sh`;
-%>
