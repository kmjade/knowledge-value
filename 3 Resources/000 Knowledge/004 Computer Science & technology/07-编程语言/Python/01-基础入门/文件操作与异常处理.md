# 分块讀取大檔案
def read_large_file(file_path, chunk_size=1024):
    """分块讀取大檔案"""
    with open(file_path, 'r', encoding='utf-8') as file:
        while True:
            chunk = file.read(chunk_size)
            if not chunk:
                break
            yield chunk

# 使用示例
for chunk in read_large_file('large_file.txt'):
    print(len(chunk))  # 處理每个數據块

# 使用生成器逐行處理
def process_lines(file_path):
    """逐行處理檔案"""
    with open(file_path, 'r', encoding='utf-8') as file:
        for line_num, line in enumerate(file, 1):
            yield line_num, line.strip()

# 處理CSV檔案
def read_csv(file_path):
    """讀取CSV檔案"""
    with open(file_path, 'r', encoding='utf-8') as file:
        for line in file:
            # 简单的CSV分割
            fields = line.strip().split(',')
            yield fields

for fields in read_csv('data.csv'):

### 檔案指针操作

#### 檔案指针基礎
```python
with open('example.txt', 'r', encoding='utf-8') as file:
    # tell() - 获取当前指针位置
    print(f"当前位置: {file.tell()}")
    
    # read(5) - 讀取5个字符
    print(file.read(5))
    print(f"当前位置: {file.tell()}")
    
    # seek(offset, whence) - 移动檔案指针
    file.seek(0)  # 回到檔案开头
    print(f"当前位置: {file.tell()}")
    
    file.seek(10)  # 跳转到第10个字符
    print(file.read(5))
    
    # whence参数: 0=檔案开头, 1=当前位置, 2=檔案末尾
    file.seek(-5, 2)  # 跳转到檔案末尾前5个字符
    print(file.read())
```

import os
import shutil
from datetime import datetime

def backup_files(source_dir, backup_dir):
    """備份檔案"""
    try:
        # 創建備份目錄
        timestamp = datetime.now().strftime('%Y%m%d_%H%M%S')
        backup_path = os.path.join(backup_dir, f'backup_{timestamp}')
        os.makedirs(backup_path, exist_ok=True)
        
        # 複製檔案
        for filename in os.listdir(source_dir):
            source_file = os.path.join(source_dir, filename)
            dest_file = os.path.join(backup_path, filename)
            
            if os.path.isfile(source_file):
                shutil.copy2(source_file, dest_file)
                print(f"備份: {filename}")
        
        print(f"備份完成: {backup_path}")
        return True
        
    except Exception as e:
        print(f"備份失败: {e}")
        return False

# 使用示例
backup_files('data', 'backups')

# 分析
```python
import re
from pathlib import Path

def analyze_log_file(log_path):
# 分析
    try:
        # 日志統計
        error_count = 0
        warning_count = 0
        error_messages = []
        
        # 正则表达式匹配
        error_pattern = re.compile(r'ERROR: (.*)')
        warning_pattern = re.compile(r'WARNING: (.*)')
        
        # 讀取日志檔案
        with open(log_path, 'r', encoding='utf-8') as file:
            for line in file:
                # 匹配错误
                error_match = error_pattern.search(line)
                if error_match:
                    error_count += 1
                    error_messages.append(error_match.group(1))
                
                # 匹配警告
                warning_match = warning_pattern.search(line)
                if warning_match:
                    warning_count += 1
        
# 分析
        return {
            'errors': error_count,
            'warnings': warning_count,
            'error_messages': error_messages[:10]  # 只返回前10个错误
        }
        
    except FileNotFoundError:
        print(f"日志檔案不存在: {log_path}")
        return None
    except Exception as e:
# 分析
        return None

# 使用示例
result = analyze_log_file('app.log')
if result:
    print(f"错误数: {result['errors']}")
    print(f"警告数: {result['warnings']}")
    print("最近错误:")
    for msg in result['error_messages']:
        print(f"  - {msg}")
```
