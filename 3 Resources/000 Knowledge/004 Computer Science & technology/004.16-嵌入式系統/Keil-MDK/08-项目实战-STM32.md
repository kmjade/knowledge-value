## 3. GPIO LED 閃爍 Blink

```c
// GPIO 初始化
void GPIO_Init_LED(void) {
    RCC->AHB1ENR |= RCC_AHB1ENR_GPIODEN;   // GPIOD 時鐘
    
    GPIOD->MODER   &= ~(3 << (12*2));       // 清除 PD12 mode
    GPIOD->MODER   |=  (1 << (12*2));       // PD12: Output
    GPIOD->OTYPER  &= ~(1 << 12);           // Push-pull
    GPIOD->OSPEEDR |=  (3 << (12*2));       // High speed
    GPIOD->PUPDR   &= ~(3 << (12*2));       // No pull
}

void LED_Toggle(void) {
    GPIOD->ODR ^= (1 << 12);    // Toggle PD12 (Green LED on Discovery)
}
```

## 4. UART printf 重定向

```c
// UART2: PA2(TX), PA3(RX) @ 115200
void UART2_Init(void) {
    RCC->AHB1ENR |= RCC_AHB1ENR_GPIOAEN;
    RCC->APB1ENR |= RCC_APB1ENR_USART2EN;
    
    GPIOA->MODER   |= (2 << (2*2)) | (2 << (3*2));  // AF mode
    GPIOA->AFR[0]  |= (7 << (2*4)) | (7 << (3*4));  // AF7 (USART2)
    
    USART2->BRR = SystemCoreClock / 115200;   // Baud rate
    USART2->CR1 = USART_CR1_TE | USART_CR1_RE | USART_CR1_UE;
}

// printf 重定向到 UART2
int fputc(int ch, FILE *f) {
    while (!(USART2->SR & USART_SR_TXE));
    USART2->DR = (uint8_t)ch;
    return ch;
}

// 使用
printf("SystemCoreClock: %lu Hz\r\n", SystemCoreClock);
```

## 5. RTX5 任務建立 RTX5 Tasks

```c
#include "cmsis_os2.h"

// RTX5 Thread 定義
osThreadId_t tid_Blink;
osThreadId_t tid_Print;

void BlinkTask(void *arg) {
    for(;;) {
        LED_Toggle();
        osDelay(500);    // 500ms
    }
}

void PrintTask(void *arg) {
    int counter = 0;
    for(;;) {
        printf("Tick: %d\r\n", counter++);
        osDelay(1000);
    }
}

// Thread 屬性
const osThreadAttr_t BlinkAttr = {
    .name       = "Blink",
    .priority   = osPriorityNormal,
    .stack_size = 200
};

int main(void) {
    SystemCoreClockUpdate();
    GPIO_Init_LED();
    UART2_Init();
    
    osKernelInitialize();
    tid_Blink = osThreadNew(BlinkTask, NULL, &BlinkAttr);
    tid_Print = osThreadNew(PrintTask, NULL, NULL);
    osKernelStart();
    for(;;);
}
```

## 6. ULINK 調試操作 Debug with ULINK

```
調試步驟 Debug Steps:
1. F7                  → Build
2. Options → Debug     → Select ULINKpro / ST-Link
3. Options → Debug     → Settings → SWD, 4 MHz
4. Ctrl+F5             → Start Debug
5. F9                  → 在 LED_Toggle() 設定斷點
6. F5                  → Run → 停在斷點
7. View → Watch 1      → 添加 GPIOD->ODR 變數
8. View → System Viewer → GPIO D → 查看寄存器值
9. View → Serial Windows → USART #2 → printf 輸出
10. F10                → Step Over 觀察變數變化
```
