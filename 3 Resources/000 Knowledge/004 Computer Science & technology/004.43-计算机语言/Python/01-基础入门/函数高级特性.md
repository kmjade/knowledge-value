## 🎨 装饰器（Decorators）

### 装饰器基礎概念

#### 什么是装饰器
```python
# 装饰器本质：接受函数作为参数，返回新函数
def simple_decorator(func):
    """简单装饰器"""
    def wrapper():
        print("装饰器：函数調用前")
        result = func()
        print("装饰器：函数調用后")
        return result
    return wrapper

@simple_decorator
def say_hello():
    print("Hello, World!")

# 等价于：say_hello = simple_decorator(say_hello)
say_hello()

# 輸出：
# 装饰器：函数調用前
# Hello, World!
# 装饰器：函数調用后
```

#### 装饰器保持函数元資訊
```python
from functools import wraps

def my_decorator(func):
    @wraps(func)  # 保留原函数的元資訊
    def wrapper(*args, **kwargs):
        """包装函数的文檔"""
        print(f"調用函数: {func.__name__}")
        return func(*args, **kwargs)
    return wrapper

@my_decorator
def add(a, b):
    """計算两个数的和"""
    return a + b

print(add.__name__)    # 輸出: add（而不是wrapper）
print(add.__doc__)     # 輸出: 計算两个数的和
```

#### 计时装饰器
```python
import time
from functools import wraps

def timer_decorator(func):
    """計算函数执行時間的装饰器"""
    @wraps(func)
    def wrapper(*args, **kwargs):
        start_time = time.time()
        result = func(*args, **kwargs)
        end_time = time.time()
        execution_time = end_time - start_time
        print(f"函数 {func.__name__} 执行時間: {execution_time:.4f}秒")
        return result
    return wrapper

@timer_decorator
def slow_function():
    """模拟耗时操作"""
    time.sleep(1)
    return "完成"

slow_function()
# 輸出：函数 slow_function 执行時間: 1.0002秒
```

```python
def log_decorator(func):
    """記錄函数調用日志的装饰器"""
    @wraps(func)
    def wrapper(*args, **kwargs):
        print(f"調用函数: {func.__name__}")
        print(f"参数: args={args}, kwargs={kwargs}")
        try:
            result = func(*args, **kwargs)
            print(f"返回值: {result}")
            return result
        except Exception as e:
            print(f"异常: {e}")
            raise
    return wrapper

@log_decorator
def divide(a, b):
```

```python
def require_permission(permission):
    """参数化装饰器：檢查使用者权限"""
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            # 模拟权限檢查
```

### 高级装饰器技術

#### 类装饰器
```python
class CountCalls:
    """类装饰器：統計函数調用次数"""
    def __init__(self, func):
        self.func = func
        self.count = 0
    
    def __call__(self, *args, **kwargs):
        self.count += 1
        print(f"函数 {self.func.__name__} 調用次数: {self.count}")
        return self.func(*args, **kwargs)

@CountCalls
def greet(name):
    """打招呼"""
    return f"Hello, {name}!"

greet("Alice")  # 調用次数: 1
greet("Bob")    # 調用次数: 2
```

#### 带参数的装饰器
```python
def repeat(times):
    """重复执行函数多次"""
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            results = []
            for _ in range(times):
                result = func(*args, **kwargs)
                results.append(result)
            return results
        return wrapper
    return decorator

@repeat(times=3)
def roll_dice():
    """掷骰子"""
    import random
    return random.randint(1, 6)

print(roll_dice())  # 例如: [4, 2, 6]
```

```python
def read_large_file(file_path, chunk_size=1024):
    """逐块讀取大檔案"""
    with open(file_path, 'r', encoding='utf-8') as file:
        while True:
            chunk = file.read(chunk_size)
            if not chunk:
                break
            yield chunk

# 逐行讀取檔案
def read_lines(file_path):
    """逐行讀取檔案"""
    with open(file_path, 'r', encoding='utf-8') as file:
        for line in file:
            yield line.strip()

# 使用示例
```

```python
def consumer():

    print("消费者等待數據...")

            item = yield
            print(f"消费者處理: {item}")
    except GeneratorExit:

def producer(consumer_coroutine):
    """生产者"""
```

```python
def make_account(balance):
    """银行帳戶闭包"""
    def deposit(amount):
        nonlocal balance
        balance += amount
        return balance
    
    def withdraw(amount):
        nonlocal balance
        if amount <= balance:
            balance -= amount
            return balance
        else:
            raise ValueError("余额不足")
    
    def get_balance():
        return balance
    
    return {
        'deposit': deposit,
        'withdraw': withdraw,
        'get_balance': get_balance
    }

# 創建帳戶
account = make_account(1000)
print(account['get_balance']())  # 1000

account['deposit'](500)
print(account['get_balance']())  # 1500

account['withdraw'](200)
print(account['get_balance']())  # 1300
```

#### 延遲計算
```python
def lazy_evaluation(expensive_func):
    """延遲計算闭包"""
    cached_value = None
    computed = False
    
    def wrapper():
        nonlocal cached_value, computed
        if not computed:
            cached_value = expensive_func()
            computed = True
        return cached_value
    
    return wrapper

# 模拟耗时計算
def expensive_computation():
    print("执行耗时計算...")
    import time
    time.sleep(1)
    return "計算結果"

# 創建延遲計算函数
lazy_func = lazy_evaluation(expensive_computation)

# 第一次調用会执行計算
print(lazy_func())  # 执行耗时計算...，計算結果

# 第二次調用直接返回缓存結果
print(lazy_func())  # 計算結果（不会列印）
```

```python
def compose(*functions):
    """函数组合：将多個函数组合成一个"""
    def composed(x):
        for func in reversed(functions):
            x = func(x)
        return x
    return composed

# 定义简单函数

multiply_by_two = lambda x: x * 2
subtract_five = lambda x: x - 5
```

```python
    @wraps(func)
    def wrapper(*args, **kwargs):
        call_count = wrapper.call_count
        wrapper.call_count += 1
        
        start_time = time.time()
        result = func(*args, **kwargs)
```

### 练习3：缓存装饰器實現
```python
from functools import wraps

def memoize(max_size=128):
    """自定义缓存装饰器"""
    def decorator(func):
        cache = {}
        
        @wraps(func)
        def wrapper(*args, **kwargs):
            # 創建缓存键
            cache_key = (args, frozenset(kwargs.items()))
            
            # 檢查缓存
            if cache_key in cache:
                return cache[cache_key]
            
            # 执行函数并缓存結果
            result = func(*args, **kwargs)
            
            # 檢查缓存大小
            if len(cache) >= max_size:
                # 简单的LRU策略：移除第一个元素
                cache.pop(next(iter(cache)))
            
            cache[cache_key] = result
            return result
        
        wrapper.cache = cache
        return wrapper
    return decorator

@memoize(max_size=10)
def expensive_function(n):
    """模拟耗时函数"""
    import time
```
