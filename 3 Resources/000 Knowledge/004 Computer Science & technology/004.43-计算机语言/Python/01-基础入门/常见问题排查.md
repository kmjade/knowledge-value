## 🔍 语法错误

### 缩进错误

#### IndentationError
```python
# ❌ 错误示例1：缩进不一致
def my_function():
print("Hello")  # 错误：缺少缩进

# ❌ 错误示例2：混合使用制表符和空格
def my_function():
    print("Hello")  # 使用空格
	print("World")   # 使用制表符，会报错

# ✅ 正确做法：使用4个空格作为缩进
def my_function():
    print("Hello")
    print("World")

# ❌ 错误示例3：意外的缩进
x = 10
    y = 20  # 错误：意外缩进

# ✅ 正确做法：不要在不需要的地方缩进
x = 10
y = 20
```

#### 解決方案

```python
# 1. 使用編輯器自動格式化
```

### 符号错误

#### 缺少引号、括号等
```python
# ❌ 缺少引号
message = Hello World  # NameError: name 'Hello' is not defined

# ✅ 正确：新增引号
message = "Hello World"

# ❌ 缺少括号
print "Hello"  # Python 2语法，Python 3会报错

# ✅ 正确：新增括号
print("Hello")

# ❌ 括号不匹配
print("Hello"  # SyntaxError: unexpected EOF while parsing

# ✅ 正确：闭合括号
print("Hello")

# ❌ 冒号缺失
if x > 0  # 缺少冒号
    print("Positive")

# ✅ 正确：新增冒号
if x > 0:
    print("Positive")
```

### 语法结构错误

#### 错误的语法使用
```python
# ❌ 使用未定义的变量
print(undefined_variable)  # NameError

# ✅ 正确：先定义变量
undefined_variable = "defined"
print(undefined_variable)

# ❌ 字符串中使用了未转义的特殊字符
path = "C:\new\file.txt"  # \n 被解釋为换行符

# ✅ 正确：使用原始字符串或转义
path = r"C:\new\file.txt"
# 或
path = "C:\\new\\file.txt"

# ❌ 關鍵字作为变量名
class = "Python"  # SyntaxError: invalid syntax

# ✅ 正确：避免使用關鍵字
course = "Python"
```

## ⚠️ 運行时错误

### 类型错误

#### TypeError常见場景
```python
# ❌ 字符串与数字直接相加
result = "10" + 5  # TypeError: can only concatenate str (not "int") to str

result = int("10") + 5

result = f"10{5}"

# ❌ 列表和元组操作混淆
my_list = [1, 2, 3]
my_list[0] = 5  # ✅ 正确

my_tuple = (1, 2, 3)
my_tuple[0] = 5  # ❌ TypeError: 'tuple' object does not support item assignment

# ✅ 正确：創建新的元组
my_tuple = (5,) + my_tuple[1:]

# ❌ 函数参数类型错误
def add(a, b):
    return a + b

add("Hello", 5)  # TypeError

# ✅ 正确：传递正确的类型
add(5, 3)  # 8
add("Hello", "World")  # "HelloWorld"
```

#### 类型错误诊断
```python
# 使用类型提示
from typing import Union

def safe_add(a: Union[int, float, str], b: Union[int, float, str]) -> Union[int, float, str]:
    """安全的加法函数"""
    try:
        return a + b
    except TypeError:
        print(f"类型错误: 無法相加 {type(a)} 和 {type(b)}")
        return None

# 使用isinstance檢查类型
def process_data(data):
    if isinstance(data, list):
        return "这是一个列表"
    elif isinstance(data, dict):
        return "这是一个字典"
    else:
        return f"未知的类型: {type(data)}"
```

### 索引错误

#### IndexError常见場景
```python
# ❌ 超出列表范围
numbers = [1, 2, 3]
print(numbers[3])  # IndexError: list index out of range

if len(numbers) > 3:
    print(numbers[3])

print(numbers[-1])  # 3

try:
    print(numbers[3])
except IndexError:
    print("索引超出范围")

# ❌ 字符串索引错误
text = "Hello"
print(text[10])  # IndexError: string index out of range

# ✅ 正确：檢查长度
if len(text) > 10:
    print(text[10])
```

### 键错误

#### KeyError常见場景
```python
# ❌ 访问不存在的键
user = {"name": "Alice", "age": 25}
print(user["email"])  # KeyError: 'email'

email = user.get("email", "未知邮箱")
```

## 🧠 逻辑错误

### 算法错误

#### 常见逻辑問題
```python
# ❌ 死循环
while True:
    print("無限循环")  # 永远不会退出

# ✅ 正确：新增退出条件
count = 0
while count < 10:
    print("循环中")
    count += 1

# ❌ 除以零
def divide(a, b):
    return a / b

result = divide(10, 0)  # ZeroDivisionError

# ✅ 正确：檢查除数
def safe_divide(a, b):
    if b == 0:
        return None
    return a / b

# ❌ 错误的递归基线
def factorial(n):
    return n * factorial(n - 1)  # 無限递归

result = factorial(5)  # RecursionError

# ✅ 正确：新增基线条件
def factorial(n):
    if n <= 1:
        return 1
    return n * factorial(n - 1)
```

### 条件判斷错误

#### 逻辑運算错误
```python
# ❌ 错误的逻辑運算
x = 10
if x > 5 or x < 8:  # 总是为True
    print("条件总是满足")

# ✅ 正确：使用and
if x > 5 and x < 8:
    print("条件正确")

# ❌ 错误的赋值与比较
x = 5
if x = 5:  # 语法错误，應該是 ==
    print("x等于5")

# ✅ 正确：使用==
if x == 5:
    print("x等于5")

# ❌ 链式比较错误
x = 10
if x > 5 and x > 15:  # 逻辑矛盾
    print("这不会执行")

# ✅ 正确：使用正确的比较
if x > 5 and x < 15:
    print("这会执行")

# ❌ 混淆is和==
a = [1, 2, 3]
b = [1, 2, 3]

if a == b:  # ✅ 內容相同
    print("內容相同")

if a is b:  # ❌ 不是同一个对象
    print("同一个对象")  # 不会执行

# ✅ 正确：明确使用is和==
if a is not b:
    print("不是同一个对象")
```

## 🚀 效能問題

### 記憶體問題

```python
numbers = [1, 2, 3, 4, 5]
for i in range(len(numbers)):

# ✅ 正确：使用副本或迭代器
numbers = [1, 2, 3, 4, 5]
for i in range(len(numbers)):

# 或
numbers = [1, 2, 3, 4, 5]
for number in numbers[:]:  # 創建副本
    print(number)

# ❌ 記憶體泄漏：持有大量对象引用
large_data = []
for i in range(1000000):
    large_data.append({"data": i * 2})  # 占用大量記憶體

# ✅ 正确：使用生成器
def generate_data(n):
    for i in range(n):
        yield {"data": i * 2}  # 逐个生成，不占用大量記憶體

# 或及时释放記憶體
large_data = []
# 使用large_data...
large_data = []  # 清空引用
```


#### 低效的循环操作
```python
# ❌ 在循环中重复計算
def inefficient_function():
    result = []
    for i in range(1000):
        x = sum(range(100))  # 每次都重新計算
        result.append(i * x)
    return result

# ✅ 正确：预計算
def efficient_function():
    result = []
    x = sum(range(100))  # 只計算一次
    for i in range(1000):
        result.append(i * x)
    return result

# ❌ 多次調用len()
data = list(range(1000))
for i in range(len(data)):
    print(data[i])

# ✅ 正确：直接迭代
for item in data:
    print(item)

# ❌ 列表推导式中的重复計算
result = [x * expensive_calculation(x) for x in range(100)]

# ✅ 正确：使用生成器表达式
result = (x * expensive_calculation(x) for x in range(100))
result = list(result)  # 如果需要列表
```

### 异常處理

#### 全面的异常處理
```python
# ✅ 處理多种异常
def safe_operation(value):
    try:
        result = int(value)
        return result * 2
    except ValueError as e:
        print(f"值错误: {e}")
        return None
    except TypeError as e:
        print(f"类型错误: {e}")
        return None
    except Exception as e:
        print(f"未知错误: {e}")
        return None

# ✅ 提供有用的错误資訊
def divide_numbers(a, b):
    try:
        return a / b
    except ZeroDivisionError:
        print(f"错误：尝试将 {a} 除以 {b}，除数不能为零")
        return None
    except TypeError:
        print(f"错误：参数类型不正确，{type(a)} 和 {type(b)}")
        return None

# ✅ 使用finally清理資源
def process_file(filename):
    file = None
    try:
        file = open(filename, 'r')
        content = file.read()
        return content
    except FileNotFoundError:
        print(f"檔案 {filename} 不存在")
        return None
    finally:
        if file:
            file.close()
            print("檔案已關閉")
```

## 💡 预防性編程

### 代碼審查清單

#### 编码最佳實踐
```python
# ✅ 1. 使用有意义的变量名
# ❌ 糟糕的命名
x = 10
y = 20

# ✅ 好的命名
base_price = 10
tax_rate = 20

# ✅ 2. 新增类型提示
from typing import List, Dict, Optional

def process_users(users: List[Dict[str, any]]) -> Optional[List[Dict]]:
    """處理使用者數據
    
    Args:
        users: 使用者列表，每个使用者是字典
    
    Returns:
        處理后的使用者列表，如果輸入無效则返回None
    """
    if not users:
        return None
    
    processed = []
    for user in users:
        processed_user = {**user, 'processed': True}
        processed.append(processed_user)
    
    return processed

# ✅ 3. 遵循PEP 8規範
# - 使用4个空格缩进
# - 行长度不超过79字符
# - 函数和类名使用描述性名称
# - 在函数之间空两行

# ✅ 4. 防御性編程
def get_value(dictionary, key, default=None):
    """安全地获取字典值"""
    if not isinstance(dictionary, dict):
        logging.warning(f"参数不是字典: {type(dictionary)}")
        return default
    
    return dictionary.get(key, default)

# ✅ 5. 輸入驗證
def calculate_discount(price, discount_rate):
    """計算折扣价格"""
    if not isinstance(price, (int, float)):
        raise TypeError("价格必须是数字")
    
    if not isinstance(discount_rate, (int, float)):
        raise TypeError("折扣率必须是数字")
    
    if price < 0:
        raise ValueError("价格不能为负数")
    
    if not 0 <= discount_rate <= 1:
        raise ValueError("折扣率必须在0到1之间")
    
    return price * (1 - discount_rate)
```
