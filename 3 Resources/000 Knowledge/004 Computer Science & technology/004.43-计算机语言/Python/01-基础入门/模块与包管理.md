## 📚 模块基礎概念

### 什么是模块
```python
# 模块定义
# 模块就是一个.py檔案，包含Python定义和语句

# 简单模块示例：my_calculator.py
def add(a, b):
    return a + b

def subtract(a, b):
    return a - b

if __name__ == "__main__":
    print("模块直接执行")
    result = add(10, 5)
    print(f"10 + 5 = {result}")
```

### 模块的優勢
- **代碼组织**: 将相關代碼分组
- **代碼复用**: 避免重复代碼
- **命名空間**: 避免命名冲突
- **可維護性**: 代碼更容易維護和擴展

## 📥 模块导入機制

### 导入方式

#### 1. 导入整个模块
```python
# 导入整个模块
import math

result = math.sqrt(16)  # 4.0
print(f"16的平方根: {result}")

# 导入自定义模块
# 假设創建了my_calculator.py檔案
import my_calculator

print(my_calculator.add(10, 5))    # 15
print(my_calculator.subtract(10, 3))  # 7
```

#### 2. 导入模块特定属性
```python
# 从模块导入特定函数
from math import sqrt, pow

print(sqrt(16))       # 4.0
print(pow(2, 3))       # 8

# 从自定义模块导入
from my_calculator import add, subtract

print(add(10, 5))    # 15
print(subtract(10, 3))  # 7
```

#### 3. 导入模块并使用别名
```python
# 使用as關鍵字为模块起别名
import math as mt

print(mt.sqrt(16))   # 4.0
print(mt.pi)            # 3.141592653589793

# 自定义模块别名
import my_calculator as calc

print(calc.add(10, 5))    # 15
print(calc.subtract(10, 3))  # 7
```

#### 4. 从模块导入所有属性
```python
# 导入模块中所有公共属性（不推荐）
from math import *

print(sqrt(16))       # 4.0
print(pow(2, 3))       # 8
print(pi)             # 3.141592653589793
```

#### 5. 导入特定名称
```python
# 只导入指定的名称
from math import sqrt as sq, pi as PI

print(sq(16))        # 4.0
print(PI)             # 3.141592653589793
```

from abc import ABC, abstractmethod

class DataSource(ABC):
    @abstractmethod
    def connect(self):
        """連接數據源"""
        pass
    
    @abstractmethod
    def read_data(self):
        """讀取數據"""
        pass

class DatabaseSource(DataSource):
    def connect(self):
        return DatabaseConnection()
    
    def read_data(self):
        return self.connection.query("SELECT * FROM users")

class FileSource(DataSource):
    def connect(self):
        return FileHandler()
    
    def read_data(self):
        with open(self.file_path) as f:
            return json.load(f)

#### 单例模式
```python
# singleton.py
class DatabaseConnection:
    """資料庫連接单例"""
    _instance = None
    
    def __new__(cls):
        if cls._instance is None:
            cls._instance = super().__new__(cls)
        return cls._instance
    
    def __init__(self):
        self.connection = self._connect()
    
    def _connect(self):
        """創建資料庫連接"""
        import psycopg2
        return psycopg2.connect(
            database="mydb",
            user="myuser",
            password="mypassword"
        )
    
    def execute_query(self, query, params=None):
        """执行SQL查詢"""
        cursor = self.connection.cursor()
        cursor.execute(query, params)
        return cursor.fetchall()
    
    def close(self):
        """關閉連接"""
        self.connection.close()

# 使用单例
db = DatabaseConnection()
results = db.execute_query("SELECT * FROM users")
db.close()
```

def reverse_string(text):
    return text[::-1]

    return ' '.join(word.capitalize() for word in text.split())

# utils/number_helper.py
def is_prime(n: int) -> bool:
    if n < 2:
        return False
    for i in range(2, int(n**0.5) + 1):
        if n % i == 0:
            return False
    return True

def factorial(n: int) -> int:
    if n < 0:
        raise ValueError("Factorial is only defined for non-negative integers")
    if n == 0 or n == 1:
        return 1
    return n * factorial(n-1)

### 模块热重载（開發除錯）
```python
# 開發时使用的热重载功能
import import importlib
import sys

def reload_module(module_name: str):
    """重新加载模块"""
    module = sys.modules.get(module_name)
    module_obj = importlib.import_module(module_name)
    importlib.reload(module_obj)

    return module_obj

# 使用示例
import simple_math
reload_module("simple_math")
```
