# 基礎语法与數據类型

> 🎯 **學習目標**：掌握Python的基礎语法体系和核心數據类型，为後續學習打下坚实基礎。

## 📑 Python语法概览

### 语言特點
- **解釋型语言**：無需编译，直接運行
- **動態类型**：变量类型在運行时確定
- **强类型语言**：不支持隐式类型转换
- **简洁优雅**：语法简洁，代碼可读性强

## 🔤 基礎數據类型

### 数字类型（ Numbers）

#### 整数（ int ）
```python
# 不同进制的表示
decimal_number = 123        # 十进制
binary_number = 0b1010      # 二进制 (10)
octal_number = 0o12         # 八进制 (10)
hexadecimal_number = 0xA    # 十六进制 (10)

# 大整数支持
big_number = 123456789012345678901234567890

# 整数運算
print(10 + 5)     # 15 (加法)
print(10 - 3)     # 7  (减法)  
print(10 * 6)     # 60 (乘法)
print(10 / 3)     # 3.333... (除法，返回浮点数)
print(10 // 3)    # 3  (除法，整数部分)
print(10 % 3)     # 1  (取模)
print(10 ** 3)    # 1000 (幂運算)
```

#### 浮点数（ float ）
```python
# 浮点数表示
pi = 3.1415926
price = 99.99
scientific_notation = 1.23e5  # 1.23 × 10^5 = 123000.0

# 浮点数精度問題
print(0.1 + 0.2)  # 0.30000000000000004 (精度误差)

# 解決方案：使用decimal模块
from decimal import Decimal
print(Decimal('0.1') + Decimal('0.2'))  # 0.3
```

#### 复数（ complex ）
```python
# 复数表示
z1 = 3 + 4j        # 創建复数
z2 = complex(1, 2)  # 等价于 1 + 2j

# 复数運算
print(z1.real)      # 3.0 (实部)
print(z1.imag)      # 4.0 (虚部)
print(abs(z1))      # 5.0 (模)

# 复数運算
print(z1 + z2)      # (4+6j)
print(z1 * z2)      # (-5+10j)
```

### 字符串（ str ）

#### 字符串創建
```python
# 不同引号表示
single_quotes = 'Hello, Python!'
double_quotes = "Hello, Python!"
triple_quotes = """Hello, Python!"""  # 可跨越多行

# 转义字符

```

#### 字符串操作
```python
text = "Hello, Python World!"

# 长度
print(len(text))                    # 21

# 大小写转换
print(text.lower())                 # hello, python world!
print(text.upper())                 # HELLO, PYTHON WORLD!
print(text.title())                 # Hello, Python World!

# 字符串尋找和替换
print(text.find('Python'))         # 7 (起始位置)
print(text.replace('World', 'Universe'))  # Hello, Python Universe!

# 字符串分割
words = text.split(', ')           # ['Hello', 'Python World!']
print(words[1])                     # Python World!

# 字符串拼接
first_name = "Python"
last_name = "Programming"
full_name = f"{first_name} {last_name}"  # f-string格式化
print(full_name)                    # Python Programming
```

#### 字符串格式化
```python
name = "Python"
version = 3.9

# %格式化
print("Language: %s, Version: %.1f" % (name, version))

# str.format()
print("Language: {}, Version: {}".format(name, version))

# f-string（推荐）
print(f"Language: {name}, Version: {version}")

# 格式化选项
number = 1234.5678
print(f"{number:,.2f}")      # 1,234.57 (千分位)
print(f"{number:.2%}")        # 123456.78% (百分比)
print(f"{number:e}")          # 1.234568e+03 (科學计数法)
```

### 列表（ list ）

#### 列表創建和操作
```python
# 創建列表
empty_list = []
numbers = [1, 2, 3, 4, 5]
mixed = [1, "hello", 3.14, True]
nested = [[1, 2], [3, 4], [5, 6]]
```

### 🗂️ 集合數據类型

### 集合（ set ）

#### 集合操作
```python
# 創建集合
empty_set = set()
numbers = {1, 2, 3, 4, 5}
unique = set([1, 2, 2, 3, 3, 3])  # {1, 2, 3}

# 集合操作
set_a = {1, 2, 3, 4}
set_b = {3, 4, 5, 6}

# 交集
intersection = set_a & set_b           # {3, 4}
intersection = set_a.intersection(set_b)

# 并集
union_set = set_a | set_b              # {1, 2, 3, 4, 5, 6}
union_set = set_a.union(set_b)

# 差集
difference = set_a - set_b             # {1, 2}
difference = set_a.difference(set_b)

# 对称差集
sym_diff = set_a ^ set_b               # {1, 2, 5, 6}
sym_diff = set_a.symmetric_difference(set_b)

# 集合應用程式
# 1. 去重
names = ['Alice', 'Bob', 'Alice', 'Charlie']
unique_names = set(names)  # {'Alice', 'Bob', 'Charlie'}

# 2. 快速尋找
valid_colors = {'red', 'green', 'blue'}
color = 'red'
if color in valid_colors:
    print(f"{color} is a valid color")
```

## 🔤 类型檢查和转换

### 类型檢查
```python
# isinstance()函数
x = 10
if isinstance(x, int):
    print("x is an integer")

# 多类型檢查
def process_data(data):
    if isinstance(data, (list, tuple)):
        return len(data)
    elif isinstance(data, str):
        return len(data)
    elif isinstance(data, dict):
        return len(data.keys())
    else:
        return 0
```

### 类型转换
```python
# 隐式转换（安全转换）
x = 1 + 2.5  # 3.5 (int自動转为float)

# 显式转换
num_str = "123"
num_int = int(num_str)        # 123 (字符串转整数)
num_float = float(num_str)    # 123.0 (字符串转浮点数)
num_str = str(num_int)        # "123" (整数转字符串)

# 布尔转换
print(bool(0))           # False
print(bool(""))          # False
print(bool([]))          # False
print(bool(None))        # False
print(bool("text"))      # True
print(bool([1, 2, 3]))   # True
```

## 🎯 實踐练习

### 练习1：学生成绩統計
```python
# 学生數據
students = {
    'Alice': {'math': 85, 'english': 92, 'science': 78},
    'Bob': {'math': 90, 'english': 88, 'science': 85},
    'Charlie': {'math': 76, 'english': 85, 'science': 90}
}

# 計算平均分
def calculate_average(grades):
    return sum(grades.values()) / len(grades)

# 找出最高分
def find_highest_score(scores):
    return max(scores.items(), key=lambda x: x[1])

# 統計結果
averages = {name: calculate_average(grades) for name, grades in students.items()}
print("班级平均分排行榜:")
sorted_averages = sorted(averages.items(), key=lambda x: x[1], reverse=True)
for rank, (name, avg) in enumerate(sorted_averages, 1):
    print(f"第{rank}名: {name}, 平均分: {avg:.2f}")
```

# 分析
```python
# 文本數據
text = """
Python is a popular programming language.
Python is known for its simplicity and readability.
Many developers love Python for its capabilities.
"""

# 分析
def analyze_text(text):
    # 转为小写并分割单词
    words = text.lower().split()
    
    # 統計单词频率
    word_count = {}
    for word in words:
        cleaned_word = word.strip('.,!?')
        word_count[cleaned_word] = word_count.get(cleaned_word, 0) + 1
    
    # 找出最常见词
    most_common = max(word_count.items(), key=lambda x: x[1])
    
    # 統計資訊
    return {
        'word_count': len(word_count),
        'character_count': len(text.replace('\n', '').replace(' ', '')),
        'most_common_word': most_common[0],
        'most_common_count': most_common[1]
    }

result = analyze_text(text)
print(f"单词总数: {result['word_count']}")
print(f"字符总数: {result['character_count']}")
print(f"最常见词: '{result['most_common_word']}' (出现{result['most_common_count']}次)")
```

## 📚 最佳實踐

### 代碼风格
```python
# ✅ 良好的代碼风格
import datetime
import json

def process_user_data(users: list) -> dict:
    """
    處理使用者數據，統計相關資訊
    
    Args:
        users: 使用者數據列表
        
    Returns:
        包含統計資訊的字典
    """
    processed_data = {}
    
    for user in users:
        user_id = user.get('id', 'unknown')
        if user_id not in processed_data:
            processed_data[user_id] = {
                'count': 0,
                'ages': [],
                'last_updated': datetime.datetime.now()
            }
        
        processed_data[user_id]['count'] += 1
        processed_data[user_id]['ages'].append(user.get('age', 0))
    
    return processed_data
```

```python
def get_type_info(variable):
    """获取变量的类型和详情資訊"""
    info = {
        'type': type(variable).__name__,
        'length': len(variable) if hasattr(variable, '__len__') else None,
        'memory_size': sys.getsizeof(variable),
        'methods': [method for method in dir(variable) if not method.startswith('_')]
    }
    
    if isinstance(variable, (list, tuple, set)):
        info['unique_elements'] = len(set(variable))
    
    return info

# 使用示例
import sys
```
