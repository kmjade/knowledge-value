## 📑 資料格式速覽

| 格式 | 特色 | 適合場景 | Python 支援 |

| CSV | 表格文字，簡單易讀 | 表格資料交換 | `csv` 模組 / pandas |

## 📄 CSV 處理

### csv 模組基礎

```python
import csv

# 讀取 CSV
with open('data.csv', newline='', encoding='utf-8') as f:
    reader = csv.reader(f)
    header = next(reader)           # 讀取標題列
    for row in reader:
        print(row)                  # row 是 list[str]

# DictReader — 以字典存取
with open('data.csv', newline='', encoding='utf-8') as f:
    reader = csv.DictReader(f)
    for row in reader:
        print(row['name'], row['age'])  # 以欄位名存取

# 寫入 CSV
with open('output.csv', 'w', newline='', encoding='utf-8') as f:
    writer = csv.writer(f)
    writer.writerow(['name', 'age', 'city'])   # 標題列
    writer.writerows([                          # 多列寫入
        ['Alice', 30, 'Taipei'],
        ['Bob', 25, 'Kaohsiung'],
    ])

# DictWriter
fieldnames = ['name', 'age', 'city']
with open('output.csv', 'w', newline='', encoding='utf-8') as f:
    writer = csv.DictWriter(f, fieldnames=fieldnames)
    writer.writeheader()
    writer.writerow({'name': 'Alice', 'age': 30, 'city': 'Taipei'})
```

### CSV 方言與特殊處理

```python
# TSV (Tab-Separated)
with open('data.tsv', newline='') as f:
    reader = csv.reader(f, delimiter='\t')

# 自訂方言
csv.register_dialect('myformat', delimiter='|', quoting=csv.QUOTE_MINIMAL)
with open('data.txt', newline='') as f:
    reader = csv.reader(f, dialect='myformat')

# 處理含換行的欄位
with open('data.csv', newline='', encoding='utf-8') as f:
    reader = csv.reader(f, quotechar='"')
```

### 進階場景

```python
import json
from datetime import datetime
from decimal import Decimal

# 自訂 encoder: 處理 datetime / Decimal
class CustomEncoder(json.JSONEncoder):
    def default(self, obj):
        if isinstance(obj, datetime):
            return obj.isoformat()
        if isinstance(obj, Decimal):
            return str(obj)
        if isinstance(obj, set):
            return list(obj)
        return super().default(obj)

data = {'created': datetime.now(), 'amount': Decimal('99.95')}
json_str = json.dumps(data, cls=CustomEncoder)
print(json_str)  # {"created": "2026-05-30T12:00:00", "amount": "99.95"}

# 大檔案串流讀取 (JSON Lines — 每行一個 JSON 物件)
def read_jsonl(path):
    with open(path, encoding='utf-8') as f:
        for line in f:
            yield json.loads(line.strip())

# JSON 合併 (merge)
def deep_merge(base: dict, override: dict) -> dict:
    for key, value in override.items():
        if key in base and isinstance(base[key], dict) and isinstance(value, dict):
            deep_merge(base[key], value)
        else:
            base[key] = value
    return base
```

## 📋 XML 處理

### xml.etree.ElementTree

```python
import xml.etree.ElementTree as ET

# 解析 XML 字串
xml_str = '''<?xml version="1.0"?>
<config>
    <database host="localhost" port="5432">
        <name>mydb</name>
    </database>
    <servers>
        <server>web01</server>
        <server>web02</server>
    </servers>
</config>'''

root = ET.fromstring(xml_str)

# 存取元素與屬性
db = root.find('database')
print(db.get('host'))          # localhost
print(db.find('name').text)    # mydb

# 遍歷
for server in root.findall('.//server'):  # XPath 風格
    print(server.text)

# 寫入 XML
tree = ET.ElementTree(root)
tree.write('config.xml', encoding='utf-8', xml_declaration=True)
```

## 🗄️ sqlite3 — 嵌入式資料庫

```python
import sqlite3

# 建立連線 (檔案不存在會自動建立)
conn = sqlite3.connect('app.db')

# Row factory — 以名稱存取欄位
conn.row_factory = sqlite3.Row

# 建立表格
conn.execute('''
    CREATE TABLE IF NOT EXISTS users (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        name TEXT NOT NULL,
        email TEXT UNIQUE,
        created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
    )
''')

# 插入
conn.execute(
    'INSERT INTO users (name, email) VALUES (?, ?)',
    ('Alice', 'alice@example.com')
)

# 批次插入
users = [('Bob', 'bob@test.com'), ('Charlie', 'charlie@test.com')]
conn.executemany('INSERT INTO users (name, email) VALUES (?, ?)', users)

# 查詢
cursor = conn.execute('SELECT * FROM users WHERE name LIKE ?', ('A%',))
for row in cursor:
    print(dict(row))  # {'id': 1, 'name': 'Alice', ...}

# 事務 (Context Manager 自動 commit/rollback)
try:
    with conn:
        conn.execute('UPDATE users SET email = ? WHERE id = ?', ('new@mail.com', 1))
except sqlite3.Error as e:
    print(f"Error: {e}")

conn.close()
```

## 🔤 檔案編碼

### 常見編碼問題

```python
# UTF-8 with BOM — Windows 常用
with open('data.csv', encoding='utf-8-sig') as f:
    content = f.read()  # BOM \ufeff 會被自動移除

# 檢測編碼 (chardet)
# import chardet
# with open('unknown.txt', 'rb') as f:
#     raw = f.read()
#     result = chardet.detect(raw)
#     print(result)  # {'encoding': 'utf-8', 'confidence': 0.99}

# 處理編碼錯誤
with open('data.txt', encoding='utf-8', errors='replace') as f:
    content = f.read()  # 無法解碼的字元替換為 �

# errors 選項
# 'strict'  — 拋出 UnicodeError (預設)
# 'ignore'  — 跳過無法解碼的字元
# 'replace' — 替換為 U+FFFD (�)
# 'xmlcharrefreplace' — &#NNNN; 實體 (寫入時)
```

### 常用編碼速查

| 編碼 | 說明 | 使用場景 |
|:---|:---|:---|
| `utf-8` | 無 BOM UTF-8 | Linux / macOS / Web 標準 |
| `utf-8-sig` | 帶 BOM UTF-8 | Windows Excel / Notepad |
| `cp950` / `big5` | 繁體中文 | 台灣舊系統 |
| `gbk` | 簡體中文 | 中國舊系統 |
| `latin-1` | ISO-8859-1 | 西歐語言 |

## 🧱 二進位處理 (struct)

```python
import struct

# 打包 (pack) — Python 值 → bytes
packed = struct.pack('>I 2s f', 1024, b'AB', 3.14)
# >  : big-endian
# I  : unsigned int (4 bytes)
# 2s : 2-byte string
# f  : float (4 bytes)
print(packed)  # b'\x00\x00\x04\x00AB@H\xf5\xc3'
print(len(packed))  # 10 bytes

# 解包 (unpack) — bytes → Python 值
value = struct.unpack('>I 2s f', packed)
print(value)  # (1024, b'AB', 3.140000104904175)

# 格式字元速查
# b/B: signed/unsigned char (1 byte)
# h/H: short (2 bytes)
# i/I: int (4 bytes)
# q/Q: long long (8 bytes)
# f/d: float/double (4/8 bytes)
# s:  string (需指定長度)
# >/<: big/little endian
```

## 📊 資料處理管線範例

```python
import pandas as pd
import json

def data_pipeline(csv_path: str, output_json: str):
    """展示完整 ETL 管線"""

    # 1. Extract — 讀取 CSV
    df = pd.read_csv(csv_path, encoding='utf-8')

    # 2. Transform — 清洗與轉換
    df['date'] = pd.to_datetime(df['date'])
    df['amount'] = pd.to_numeric(df['amount'], errors='coerce')
    df = df.dropna(subset=['amount'])           # 移除缺失值
    df['category'] = df['category'].str.strip().str.title()  # 標準化

    # 3. Aggregate — 彙總
    summary = df.groupby('category').agg(
        total=('amount', 'sum'),
        avg=('amount', 'mean'),
        count=('amount', 'count'),
    ).reset_index()

    # 4. Load — 寫入 JSON
    result = summary.to_dict(orient='records')
    with open(output_json, 'w', encoding='utf-8') as f:
        json.dump(result, f, indent=2, ensure_ascii=False)

    print(f"Processed {len(df)} rows → {len(result)} categories")
    return result
```

## 🎯 重點回顧

1. **CSV** — 用 `csv.DictReader`/`DictWriter`，注意 `newline=''` 和 `encoding`
2. **JSON** — `json.load/dump`，客製化 `JSONEncoder` 處理特殊型別
3. **XML** — `xml.etree.ElementTree` 基礎；大規模用 `lxml`
4. **sqlite3** — 嵌入式關聯式，`?` 佔位符防 SQL injection，`row_factory` 便利存取
5. **pandas** — DataFrame / groupby / merge 三核心；I/O 支援多格式
6. **numpy** — 向量化運算、廣播、多維陣列
7. **編碼** — `utf-8-sig` 處理 BOM，`errors='replace'` 容錯
8. **二進位** — `struct.pack/unpack` 處理 binary protocol

---

> 💡 **最佳實踐**：CSV 用 pandas 更簡潔；JSON 大檔案用串流 (ijson)；ETL 用 pandas 管線 + sqlite3；編碼一律顯式指定 `encoding='utf-8'`。
