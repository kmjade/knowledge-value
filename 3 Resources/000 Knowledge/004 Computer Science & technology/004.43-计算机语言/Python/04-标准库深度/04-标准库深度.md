## 📦 collections — 進階容器

### namedtuple

```python
from collections import namedtuple

# 定義輕量資料型別（不可變，可像 tuple 一樣解構）
Point = namedtuple('Point', ['x', 'y'])
p = Point(10, 20)
print(p.x, p.y)          # 10 20
print(p[0], p[1])        # 10 20
print(p._asdict())       # {'x': 10, 'y': 20}

# 帶預設值 (Python 3.7+)
Employee = namedtuple('Employee', ['name', 'dept'], defaults=['Engineering'])
e = Employee('Alice')
print(e)  # Employee(name='Alice', dept='Engineering')
```

### deque — 雙端佇列

```python
from collections import deque

dq = deque([1, 2, 3], maxlen=5)  # 固定長度 FIFO 緩衝

dq.append(4)       # 右端加入: [1, 2, 3, 4]
dq.appendleft(0)   # 左端加入: [0, 1, 2, 3, 4]
dq.pop()           # 右端彈出 → 4
dq.popleft()       # 左端彈出 → 0
dq.extend([5, 6])  # 右端擴展

# O(1) 旋轉
dq.rotate(1)       # 右旋 [3, 1, 2]
dq.rotate(-1)      # 左旋 [1, 2, 3]
```

### Counter — 計數器

```python
from collections import Counter

words = ['red', 'blue', 'red', 'green', 'blue', 'blue']
cnt = Counter(words)
print(cnt)                    # Counter({'blue': 3, 'red': 2, 'green': 1})
print(cnt.most_common(2))     # [('blue', 3), ('red', 2)]
print(cnt['red'])             # 2

# 運算
c1 = Counter(a=3, b=1)
c2 = Counter(a=1, b=2)
print(c1 + c2)   # Counter({'a': 4, 'b': 3})
print(c1 - c2)   # Counter({'a': 2})
print(c1 & c2)   # 交集: Counter({'a': 1, 'b': 1})
print(c1 | c2)   # 聯集: Counter({'a': 3, 'b': 2})
```

### defaultdict

```python
from collections import defaultdict

# 避免 KeyError，自動提供預設值
dd = defaultdict(list)
dd['fruits'].append('apple')
dd['fruits'].append('banana')
print(dd)  # defaultdict(<class 'list'>, {'fruits': ['apple', 'banana']})

# 常用預設型別
by_type = defaultdict(set)     # 分組去重
counts = defaultdict(int)      # 計數器
tree = defaultdict(dict)       # 嵌套結構

# 自訂預設值
items = defaultdict(lambda: {'count': 0, 'total': 0.0})
items['A']['count'] += 1
```

### ChainMap — 合併多個映射

```python
from collections import ChainMap

defaults = {'host': 'localhost', 'port': 5432}
cli_args = {'host': 'prod.example.com'}

config = ChainMap(cli_args, env_vars, defaults)

```

## 🔁 itertools — 迭代器工具

### 無限迭代器

```python
from itertools import count, cycle, repeat

# count(start, step) — 無窮計數器
for i, v in zip(range(5), count(10, 2)):
    print(v, end=' ')  # 10 12 14 16 18

# cycle(iterable) — 無限循環
# for c in cycle('ABC'):  # A B C A B C ...

# repeat(obj, times) — 重複
print(list(repeat('Hi', 3)))  # ['Hi', 'Hi', 'Hi']
```

### 組合迭代器

```python
from itertools import product, permutations, combinations, combinations_with_replacement

items = ['A', 'B', 'C']

# product — 笛卡爾積
print(list(product(items, repeat=2)))
# [('A','A'), ('A','B'), ('A','C'), ('B','A'), ...]

# permutations — 排列 (n! / (n-r)!)
print(list(permutations(items, 2)))
# [('A','B'), ('A','C'), ('B','A'), ('B','C'), ('C','A'), ('C','B')]

# combinations — 組合 (n! / (r!(n-r)!))
print(list(combinations(items, 2)))
# [('A','B'), ('A','C'), ('B','C')]

# combinations_with_replacement — 可重複組合
print(list(combinations_with_replacement(items, 2)))
```

### 實用迭代器

```python
from itertools import chain, groupby, accumulate, islice, pairwise

# chain — 串接多個可迭代物件
print(list(chain('AB', 'CD', [1, 2])))  # ['A', 'B', 'C', 'D', 1, 2]

# groupby — 按 key 分組（需先排序）
data = [('A', 1), ('A', 2), ('B', 3), ('B', 4)]
for key, group in groupby(data, key=lambda x: x[0]):
    print(key, list(group))
# A [('A', 1), ('A', 2)]
# B [('B', 3), ('B', 4)]

# accumulate — 累積運算
import operator
print(list(accumulate([1, 2, 3, 4])))              # [1, 3, 6, 10]
print(list(accumulate([1, 2, 3, 4], operator.mul))) # [1, 2, 6, 24]

# pairwise (Python 3.10+) — 滑動窗口
print(list(pairwise([1, 2, 3, 4])))  # [(1,2), (2,3), (3,4)]
```

## 🧰 functools — 高階函數工具

### lru_cache

```python
from functools import lru_cache

@lru_cache(maxsize=128)
def fibonacci(n: int) -> int:
    """帶緩存的遞迴 Fibonacci"""
    if n < 2:
        return n
    return fibonacci(n - 1) + fibonacci(n - 2)

print(fibonacci(100))          # 354224848179261915075 (瞬間完成)
print(fibonacci.cache_info())  # CacheInfo(hits=98, misses=101, maxsize=128, currsize=101)
fibonacci.cache_clear()        # 清除緩存
```

### partial — 偏函數

```python
from functools import partial

# 固定部分參數
def power(base, exp):
    return base ** exp

square = partial(power, exp=2)
cube   = partial(power, exp=3)

print(square(5))  # 25
print(cube(5))    # 125

# 實用範例：固定 open 的 encoding
open_utf8 = partial(open, encoding='utf-8')
```

### reduce

```python
from functools import reduce

# 將序列歸約為單一值
numbers = [1, 2, 3, 4, 5]
print(reduce(lambda acc, x: acc * x, numbers))     # 120 (5!)
print(reduce(lambda acc, x: acc * x, numbers, 10))  # 1200 (10 * 5!)

# 實用：深層字典取值
from functools import reduce
import operator

def deep_get(d, keys):
    return reduce(operator.getitem, keys, d)

data = {'a': {'b': {'c': 42}}}
print(deep_get(data, ['a', 'b', 'c']))  # 42
```

### singledispatch

```python
from functools import singledispatch

@singledispatch
def format_value(value):
    """泛型函數，按型別分派"""
    return str(value)

@format_value.register
def _(value: int):
    return f"0x{value:x}"

@format_value.register
def _(value: list):
    return f"[{', '.join(format_value(v) for v in value)}]"

print(format_value(255))           # 0xff
print(format_value([1, 2, 3]))     # [0x1, 0x2, 0x3]
print(format_value(3.14))          # 3.14 (fallback)
```

## 📅 datetime / zoneinfo

```python
from datetime import datetime, date, time, timedelta, timezone
from zoneinfo import ZoneInfo

# 基本操作
now = datetime.now()
today = date.today()
print(now.isoformat())       # 2026-05-30T12:00:00

# timedelta 運算
tomorrow = now + timedelta(days=1)
one_hour_ago = now - timedelta(hours=1)

# 格式化
print(now.strftime('%Y-%m-%d %H:%M:%S'))   # strftime
dt = datetime.strptime('2026-05-30', '%Y-%m-%d')  # strptime

# 時區處理 (zoneinfo, Python 3.9+)
tokyo_tz = ZoneInfo('Asia/Tokyo')
ny_tz    = ZoneInfo('America/New_York')

tokyo_time = datetime.now(tokyo_tz)
ny_time    = datetime.now(ny_tz)
print(f"Tokyo: {tokyo_time}, NY: {ny_time}")

# UTC 時間戳
ts = datetime.now(timezone.utc).timestamp()
```

## 🔍 re — 正則表達式 Cheatsheet

| 語法 | 含義 | 範例 |
|:---|:---|:---|
| `.` | 任意字元 (不含換行) | `a.b` → `aab`, `acb` |
| `*` | 0 或多次 | `ab*` → `a`, `ab`, `abb` |
| `+` | 1 或多次 | `ab+` → `ab`, `abb` |
| `?` | 0 或 1 次 (非貪婪) | `ab?` → `a`, `ab` |
| `{m,n}` | m 到 n 次 | `a{2,4}` → `aa`, `aaa`, `aaaa` |
| `^` / `$` | 行首 / 行尾 | `^Hello` |
| `\d` / `\D` | 數字 / 非數字 | `\d{3}` → `123` |
| `\w` / `\W` | 單字字元 / 非單字 | `\w+` |
| `\s` / `\S` | 空白 / 非空白 | `\s+` |
| `[abc]` | 字元類 | `[aeiou]` |
| `[^abc]` | 否定字元類 | `[^0-9]` |
| `( ... )` | 捕獲群組 | `(\d{4})-(\d{2})` |
| `(?: ... )` | 非捕獲群組 | `(?:https?://)?` |
| `(?P<name>...)` | 命名群組 | `(?P<year>\d{4})` |
| `(?= ... )` | 正向 lookahead | `foo(?=bar)` |
| `(?! ... )` | 負向 lookahead | `foo(?!bar)` |

### 常用函數

```python
import re

text = "Contact: alice@example.com, bob@test.org"

# search — 找到第一個匹配
match = re.search(r'[\w.-]+@[\w.-]+\.\w+', text)
print(match.group())  # alice@example.com

# findall — 找到所有匹配
emails = re.findall(r'[\w.-]+@[\w.-]+\.\w+', text)
print(emails)  # ['alice@example.com', 'bob@test.org']

# sub — 替換
masked = re.sub(r'[\w.-]+@([\w.-]+)', r'***@\1', text)
print(masked)  # Contact: ***@example.com, ***@test.org

# 預先編譯 (效能優化)
EMAIL_RE = re.compile(r'[\w.-]+@[\w.-]+\.\w+', re.IGNORECASE)
print(EMAIL_RE.findall(text))

# 命名群組
date_re = re.compile(r'(?P<year>\d{4})-(?P<month>\d{2})-(?P<day>\d{2})')
m = date_re.search('2026-05-30')
print(m.group('year'))   # 2026
print(m.groupdict())     # {'year': '2026', 'month': '05', 'day': '30'}
```

## 📝 logging — 結構化日誌

```python
import logging

# 基本配置
logging.basicConfig(
    level=logging.INFO,
    format='%(asctime)s [%(levelname)s] %(name)s: %(message)s',
    datefmt='%Y-%m-%d %H:%M:%S',
    handlers=[
        logging.FileHandler('app.log'),
        logging.StreamHandler()
    ]
)

logger = logging.getLogger(__name__)
logger.debug("Debug message")   # 不輸出 (level=INFO)
logger.info("Server started")   # 輸出
logger.warning("Disk 80% full") # 輸出
logger.error("Connection failed")
logger.exception("Fatal error") # 附帶 traceback

# 層級：DEBUG(10) < INFO(20) < WARNING(30) < ERROR(40) < CRITICAL(50)
```

### Logger 層級結構

```python
# 父-子 logger 繼承設定
root_logger = logging.getLogger()       # root
app_logger  = logging.getLogger('app')  # 繼承 root
mod_logger  = logging.getLogger('app.module')  # 繼承 app

# 自訂 Handler
import sys

handler.setLevel(logging.WARNING)
handler.setFormatter(logging.Formatter('%(levelname)s - %(message)s'))
mod_logger.addHandler(handler)
```

## 🏷️ argparse — CLI 參數解析

```python
# cli.py
import argparse

def main():
    parser = argparse.ArgumentParser(
        description='Process some files.',
        epilog='Example: cli.py -i input.txt -o output.txt'
    )

    # 位置參數
    parser.add_argument('files', nargs='+', help='input files')

    # 可選參數
    parser.add_argument('-o', '--output', default='output.txt',
                        help='output file (default: output.txt)')
    parser.add_argument('-v', '--verbose', action='store_true',
                        help='enable verbose output')
    parser.add_argument('-n', '--count', type=int, default=1,
                        help='repeat count')
    parser.add_argument('--mode', choices=['fast', 'safe'], default='safe')

    # 互斥群組
    group = parser.add_mutually_exclusive_group()
    group.add_argument('--compress', action='store_true')
    group.add_argument('--decompress', action='store_true')

    args = parser.parse_args()

    if args.verbose:
        print(f"Processing {len(args.files)} files...")
        print(f"Output: {args.output}, Count: {args.count}, Mode: {args.mode}")

if __name__ == '__main__':
    main()
# Usage: python cli.py file1 file2 -o out.txt -v -n 3 --mode fast
```

## 🎯 重點回顧

1. **`collections`** 提供 `Counter` / `deque` / `defaultdict` / `ChainMap` 等專業容器
2. **`itertools`** 的 `product` / `permutations` / `groupby` 是組合運算利器
3. **`functools.lru_cache`** 快速實現記憶化 (memoization)
4. **`pathlib`** 是現代化檔案路徑 API，取代 `os.path`
5. **`zoneinfo`** (3.9+) 正確處理時區，取代 `pytz`
6. **`logging`** 分層級、支援 handler/formatter/filter
7. **`argparse`** 快速構建 CLI；複雜需求考慮 `click` 或 `typer`

---

> 💡 **最佳實踐**：善用 `pathlib.Path` 取代字串拼接；`re.compile()` 預編譯正則提升效能；logger 按模組命名 (`__name__`) 方便追蹤。
