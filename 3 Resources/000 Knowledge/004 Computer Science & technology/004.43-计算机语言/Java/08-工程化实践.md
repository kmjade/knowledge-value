## 8.1 構建工具 Build Tools

### Maven vs Gradle

| 特性 | Maven | Gradle |
|------|-------|--------|
| **配置語言** | XML (pom.xml) | Groovy/Kotlin DSL |
| **建構速度** | 較慢 (無增量) | 快 (增量建構 + Daemon) |
| **依賴管理** | 聲明式，依賴傳遞 | 聲明式，更靈活的排除 |
| **生態** | 最成熟，插件豐富 | 成長快，Android 官方 |
| **靈活性** | 約定優於配置 (僵化) | 高度可程式化 |

### Maven 核心概念

```xml
<!-- pom.xml 範例 -->
<project>
    <groupId>com.example</groupId>
    <artifactId>my-app</artifactId>
    <version>1.0.0</version>
    <packaging>jar</packaging>

    <!-- 屬性集中管理 -->
    <properties>
        <java.version>21</java.version>
        <spring-boot.version>3.3.0</spring-boot.version>
    </properties>

    <!-- 依賴管理 (BOM) -->
    <dependencyManagement>
        <dependencies>
            <dependency>
                <groupId>org.springframework.boot</groupId>
                <artifactId>spring-boot-dependencies</artifactId>
                <version>${spring-boot.version}</version>
                <type>pom</type>
                <scope>import</scope>
            </dependency>
        </dependencies>
    </dependencyManagement>

    <dependencies>
        <dependency>
            <groupId>org.springframework.boot</groupId>
            <artifactId>spring-boot-starter-web</artifactId>
        </dependency>
        <dependency>
            <groupId>org.projectlombok</groupId>
            <artifactId>lombok</artifactId>
            <optional>true</optional>
        </dependency>
    </dependencies>
</project>
```

```bash
# Maven 生命週期
mvn clean       # 清理
mvn compile     # 編譯
mvn test        # 測試
mvn package     # 打包 (JAR/WAR)
mvn install     # 安裝到本地倉庫
mvn deploy      # 發布到遠端倉庫

# 常用命令
mvn dependency:tree                  # 依賴樹
mvn versions:display-dependency-updates  # 依賴更新
mvn clean package -DskipTests        # 跳過測試打包
mvn spring-boot:run                  # 運行 Spring Boot
```

### Gradle 核心概念

```kotlin
// build.gradle.kts (Kotlin DSL — 推薦)
plugins {
    id("org.springframework.boot") version "3.3.0"
    id("io.spring.dependency-management") version "1.1.5"
    java
}

group = "com.example"
version = "1.0.0"

java {
    sourceCompatibility = JavaVersion.VERSION_21
}

repositories {
    mavenCentral()
}

dependencies {
    implementation("org.springframework.boot:spring-boot-starter-web")
    compileOnly("org.projectlombok:lombok")
    annotationProcessor("org.projectlombok:lombok")
    testImplementation("org.springframework.boot:spring-boot-starter-test")
}

// 自定義任務
tasks.register("hello") {
    doLast { println("Hello, Gradle!") }
}
```

```bash
# Gradle 常用命令
./gradlew build        # 建構 (Linux/Mac)
gradlew.bat build      # 建構 (Windows)
./gradlew bootRun      # 運行 Spring Boot
./gradlew test         # 測試
./gradlew dependencies # 依賴報告
```

## 8.2 單元測試 Testing

### JUnit 5 核心

```java
import org.junit.jupiter.api.*;
import static org.junit.jupiter.api.Assertions.*;

class CalculatorTest {

    private Calculator calc;

    @BeforeAll
    static void initAll() { /* 所有測試前執行一次 */ }

    @BeforeEach
    void init() {
        calc = new Calculator();
    }

    @Test
    @DisplayName("加法測試")
    void testAdd() {
        assertEquals(5, calc.add(2, 3), "2 + 3 應等於 5");
    }

    @Test
    @DisplayName("除法 — 除零應拋例外")
    void testDivideByZero() {
        assertThrows(ArithmeticException.class,
            () -> calc.divide(10, 0));
    }

    @ParameterizedTest
    @ValueSource(ints = {1, 3, 5, -3, 15})
    @DisplayName("參數化 — 奇數判斷")
    void testIsOdd(int number) {
        assertTrue(calc.isOdd(number));
    }

    @RepeatedTest(3)
    @DisplayName("重複測試 — 隨機數")
    void testRandom(RepetitionInfo info) {
        assertTrue(calc.random() >= 0);
    }

    @Test
    @Timeout(value = 1, unit = TimeUnit.SECONDS)
    @DisplayName("超時測試")
    void testTimeout() {
        // 必須在 1 秒內完成
    }

    @AfterEach
    void tearDown() { /* 每個測試後執行 */ }

    @AfterAll
    static void tearDownAll() { /* 所有測試後執行一次 */ }
}
```

### Mockito — Mock 框架

```java
import static org.mockito.Mockito.*;
import static org.mockito.BDDMockito.*;

@ExtendWith(MockitoExtension.class)
class UserServiceTest {

    @Mock
    private UserRepository userRepository;

    @InjectMocks
    private UserService userService;

    @Test
    void testFindUser() {
        // Arrange
        User mockUser = new User(1L, "Alice");
        given(userRepository.findById(1L)).willReturn(Optional.of(mockUser));

        // Act
        User result = userService.findById(1L);

        // Assert
        assertThat(result.getName()).isEqualTo("Alice");
        then(userRepository).should(times(1)).findById(1L);
    }

    @Test
    void testSaveUser() {
        User user = new User("Bob");
        when(userRepository.save(any(User.class))).thenReturn(user);

        userService.createUser("Bob");

        verify(userRepository).save(argThat(u -> u.getName().equals("Bob")));
    }
}
```

### AssertJ — 流暢斷言

```java
import static org.assertj.core.api.Assertions.*;

// 流暢 API，IDE 自動補全友好
assertThat(actual)
    .isNotNull()
    .startsWith("Java")
    .contains("21")
    .endsWith("LTS");

assertThat(list)
    .hasSize(3)
    .contains("Java", "Python")
    .doesNotContainNull()
    .allMatch(s -> s.length() > 2);

// 異常用斷言
assertThatThrownBy(() -> service.process(null))
    .isInstanceOf(IllegalArgumentException.class)
    .hasMessageContaining("must not be null");
```

### 分層架構 (Controller → Service → Repository)

```java
// Controller
@RestController
@RequestMapping("/api/users")
public class UserController {

    private final UserService userService;

    public UserController(UserService userService) {
        this.userService = userService;
    }

    @GetMapping("/{id}")
    public ResponseEntity<UserDto> getById(@PathVariable Long id) {
        return ResponseEntity.ok(userService.findById(id));
    }

    @PostMapping
    @ResponseStatus(HttpStatus.CREATED)
    public UserDto create(@Valid @RequestBody CreateUserRequest req) {
        return userService.create(req);
    }

    @ExceptionHandler(UserNotFoundException.class)
    public ResponseEntity<ErrorResponse> handleNotFound(UserNotFoundException e) {
        return ResponseEntity.status(404).body(new ErrorResponse(e.getMessage()));
    }
}

// Service
@Service
@Transactional
public class UserService {
    private final UserRepository repo;

    public UserDto findById(Long id) {
        return repo.findById(id)
            .map(UserDto::from)
            .orElseThrow(() -> new UserNotFoundException(id));
    }

    public UserDto create(CreateUserRequest req) {
        User user = new User(req.name(), req.email());
        return UserDto.from(repo.save(user));
    }
}

// Repository
@Repository
public interface UserRepository extends JpaRepository<User, Long> {
    Optional<User> findByEmail(String email);

    @Query("SELECT u FROM User u WHERE u.status = :status")
    List<User> findByStatus(@Param("status") String status);
}
```

### 全域異常處理

```java
@RestControllerAdvice
public class GlobalExceptionHandler {

    @ExceptionHandler(MethodArgumentNotValidException.class)
    public ResponseEntity<Map<String, String>> handleValidation(MethodArgumentNotValidException e) {
        Map<String, String> errors = new HashMap<>();
        e.getBindingResult().getFieldErrors()
            .forEach(err -> errors.put(err.getField(), err.getDefaultMessage()));
        return ResponseEntity.badRequest().body(errors);
    }

    @ExceptionHandler(Exception.class)
    public ResponseEntity<String> handleAll(Exception e) {
        log.error("Unexpected error", e);
        return ResponseEntity.status(500).body("Internal Server Error");
    }
}
```

## 8.4 Lombok 與常用工具庫

```java
// Lombok — 減少樣板代碼
@Data                    // @Getter + @Setter + @ToString + @EqualsAndHashCode
@Builder                 // Builder 模式

@AllArgsConstructor      // 全參建構子

    private Long id;
    private String name;
    @Builder.Default

    UserDto toDto(User user);

    @Mapping(target = "id", ignore = true)

// Guava — Google 核心庫
Cache<String, User> cache = CacheBuilder.newBuilder()
```

## 8.6 Docker 容器化部署

```dockerfile
# Multi-stage Build
FROM maven:3.9-amazoncorretto-21 AS build
WORKDIR /app
COPY pom.xml .
RUN mvn dependency:go-offline
COPY src ./src
RUN mvn package -DskipTests

FROM amazoncorretto:21-alpine
WORKDIR /app
COPY --from=build /app/target/*.jar app.jar

# JVM 參數 (容器環境)
ENV JAVA_OPTS="-XX:+UseContainerSupport -XX:MaxRAMPercentage=75.0"

EXPOSE 8080
ENTRYPOINT ["sh", "-c", "java $JAVA_OPTS -jar app.jar"]
```

```yaml
# docker-compose.yml
version: '3.8'
services:
  app:
    build: .
    ports:
      - "8080:8080"
    environment:
      - SPRING_PROFILES_ACTIVE=docker
      - DB_URL=jdbc:postgresql://db:5432/mydb
    depends_on:
      db:
        condition: service_healthy
  db:
    image: postgres:16-alpine
    environment:
      POSTGRES_DB: mydb
      POSTGRES_PASSWORD: secret
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U postgres"]
```
