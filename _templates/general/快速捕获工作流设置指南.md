```javascript
<%*
// Quick capture PARA classification
const questions = [
  "这个想法需要行动吗？(是/否)",
  "有截止日期吗？(是/否)",
  "这是参考资料吗？(是/否)"
  "这是纯想法吗？(是/否)"
];

let choice = "";

// 决策树逻辑
function determinePARA(answers) {
  // 是否需要行动？
  const needsAction = await tp.system.suggester(["是", "否"], "这个想法需要行动吗？(是/否)??");

  // 有截止日期？
  const hasDeadline = await tp.system.suggester(["是", "否"], "有截止日期吗？(是/否)??");
  // 是参考资料？
  const isReference = await tp.system.suggester(["是", "否"], "这是参考资料吗？(是/否)??");
  // 是纯想法？
  const isPureIdea = await tp.system.suggester(["是", "否"], "这是纯想法吗？(是/否)??");

  // 是否有参考价值？
  const hasValue = await tp.system.suggester(["是", "否"], "这个想法有参考价值吗？(是/否)??");

  // 是否过时？
  const isOutdated = await tp.system.suggester(["是", "否"], "这个想法已经过时了吗？(是/否)??");

  // 简单选决策
  if (needsAction && hasDeadline) {
# 專案
  } else if (hasDeadline && !needsAction) {
# 專案
  } else if (isReference) {
# 資源
  } else if (isPureIdea) {
    choice = "areas"; // 纯想法 → 领域（可能需要沉思）
  } else {
    choice = "areas"; // 有价值但不需要立即行动 → 领域
  }
  }

  return `para: ${choice}`;
}
%>
```

# 新增

**脚本名称**：`移动到PARA目标`

```javascript
<%*
// Move to PARA destination based on choice
const paraChoice = tp.frontmatter.para;

const destinations = {
  "projects": "1 Projects",
  "areas": "2 Areas",
  "resources": "3 Resources",
  "archives": "4 Archives"
};

// 等待 Templater 脚本执行
const waitForTemplater = 1000; // 毫秒检查一次

setTimeout(async () => {
  while (true) {
    await new Promise(r => setTimeout(r, waitForTemplater));
    // 检查 Templater 是否已准备好
    if (tp.file.getFrontmatter()) {
      // 如果 frontmatter 中没有 para，使用脚本移动
      const paraValue = tp.frontmatter.para || paraChoice;
      const destFolder = destinations[paraValue];
      if (destFolder) {
        const destPath = await tp.app.vault.adapter.getAbstractFileByPath(destFolder);
        if (destPath) {
          await tp.app.workspace.activeLeaf.rename(
            tp.file.path,
            `${destFolder}/${tp.file.basename}`
          );
          tp.notice.success(`已移动到: ${destFolder}`);
          break;
        }
      }
    }
  }
  }
}, 5000); // 5秒后超时
})();

%>
```
